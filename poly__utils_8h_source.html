87 template <int dim, int spacedim>
88 void
89 get_face_connectivity_of_cells(
90 const parallel::fullydistributed::Triangulation<dim, spacedim>
91 &triangulation,
92 DynamicSparsityPattern &cell_connectivity,
93 const std::vector<types::global_cell_index> locally_owned_cells)
94 {
95 cell_connectivity.reinit(triangulation.n_locally_owned_active_cells(),
96 triangulation.n_locally_owned_active_cells());
97
98
99 // loop over all cells and their neighbors to build the sparsity
100 // pattern. note that it's a bit hard to enter all the connections when
101 // a neighbor has children since we would need to find out which of its
102 // children is adjacent to the current cell. this problem can be omitted
103 // if we only do something if the neighbor has no children -- in that
104 // case it is either on the same or a coarser level than we are. in
105 // return, we have to add entries in both directions for both cells
106 for (const auto &cell : triangulation.active_cell_iterators())
107 {
108 if (cell->is_locally_owned())
109 {
110 const unsigned int index = cell->active_cell_index();
111 cell_connectivity.add(get_index(locally_owned_cells, index),
112 get_index(locally_owned_cells, index));
113 for (auto f : cell->face_indices())
114 if ((cell->at_boundary(f) == false) &&
115 (cell->neighbor(f)->has_children() == false) &&
116 cell->neighbor(f)->is_locally_owned())
117 {
118 const unsigned int other_index =
119 cell->neighbor(f)->active_cell_index();
120
121 cell_connectivity.add(get_index(locally_owned_cells, index),
122 get_index(locally_owned_cells,
123 other_index));
124 cell_connectivity.add(get_index(locally_owned_cells,
125 other_index),
126 get_index(locally_owned_cells, index));
127 }
128 }
129 }

251 template <typename Value,
252 typename Options,
253 typename Translator,
254 typename Box,
255 typename Allocators>
256 void
257 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
258 const Rtree_visitor::InternalNode &node)
259 {
260 using elements_type =
261 typename boost::geometry::index::detail::rtree::elements_type<
262 InternalNode>::type; // pairs of bounding box and pointer to child node
263 const elements_type &elements =
264 boost::geometry::index::detail::rtree::elements(node);
265
266 if (level < target_level)
267 {
268 size_t level_backup = level;
269 ++level;
270
271 for (typename elements_type::const_iterator it = elements.begin();
272 it != elements.end();
273 ++it)
274 {
275 boost::geometry::index::detail::rtree::apply_visitor(*this,
276 *it->second);
277 }
278
279 level = level_backup;
280 }
281 else if (level == target_level)
282 {
283 // const unsigned int n_children = elements.size();
284 const auto offset = agglomerates.size();
285 agglomerates.resize(offset + 1);
286 row_ptr.resize(row_ptr.size() + 1);
287 next_level_leafs_processed = 0;
288 row_ptr.back().push_back(
289 next_level_leafs_processed); // convention: row_ptr[0]=0
290 size_t level_backup = level;
291
292 ++level;
293 for (const auto &child : elements)
294 {
295 boost::geometry::index::detail::rtree::apply_visitor(*this,
296 *child.second);
297 }
298 // Done with node number 'node_counter'
299
300 ++node_counter; // visited all children of an internal node
301
302 level = level_backup;
303 }
304 else if (level > target_level)
305 {
306 // Keep visiting until you go to the leafs.
307 size_t level_backup = level;
308
309 ++level;
310
311 for (const auto &child : elements)
312 {
313 boost::geometry::index::detail::rtree::apply_visitor(*this,
314 *child.second);
315 }
316 level = level_backup;
317 row_ptr[node_counter].push_back(next_level_leafs_processed);
318 }
319 }

324 typename Options,
325 typename Translator,

327 typename Allocators>
328 void
329 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
330 const Rtree_visitor::Leaf &leaf)
331 {
332 using elements_type =
333 typename boost::geometry::index::detail::rtree::elements_type<
334 Leaf>::type; // pairs of bounding box and pointer to child node
335 const elements_type &elements =
336 boost::geometry::index::detail::rtree::elements(leaf);
337
338
339 for (const auto &it : elements)
340 {
341 agglomerates[node_counter].push_back(it.second);
342 }
343 next_level_leafs_processed += elements.size();
344 }

346 template <typename Rtree>
347 inline std::pair<
348 std::vector<std::vector<unsigned int>>,
349 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
350 typename Rtree::indexable_type>::value>::active_cell_iterator>>>
351 extract_children_of_level(const Rtree &tree, const unsigned int level)
352 {
353 using RtreeView =
354 boost::geometry::index::detail::rtree::utilities::view<Rtree>;
355 RtreeView rtv(tree);
356
357 std::vector<std::vector<unsigned int>> csrs;
358 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
359 typename Rtree::indexable_type>::value>::active_cell_iterator>>
360 agglomerates;
361
362 if (rtv.depth() == 0)
363 {
364 // The below algorithm does not work for `rtv.depth()==0`, which might
365 // happen if the number entries in the tree is too small.
366 // In this case, simply return a single bounding box.
367 agglomerates.resize(1);
368 agglomerates[0].resize(1);
369 csrs.resize(1);
370 csrs[0].resize(1);
371 }
372 else
373 {
374 const unsigned int target_level =
375 std::min<unsigned int>(level, rtv.depth());
376
377 Rtree_visitor<typename RtreeView::value_type,
378 typename RtreeView::options_type,
379 typename RtreeView::translator_type,
380 typename RtreeView::box_type,
381 typename RtreeView::allocators_type>
382 node_visitor(rtv.translator(), target_level, agglomerates, csrs);
383 rtv.apply_visitor(node_visitor);
384 }
385 AssertDimension(agglomerates.size(), csrs.size());
386
387 return {csrs, agglomerates};
388 }

391 template <int dim, typename Number = double>
392 Number
393 compute_h_orthogonal(
394 const unsigned int face_index,
395 const std::vector<typename Triangulation<dim>::active_face_iterator>
396 &polygon_boundary,
397 const Tensor<1, dim> &deal_normal)
398 {
399#ifdef DEAL_II_WITH_CGAL
400
401 using Kernel = CGAL::Exact_predicates_exact_constructions_kernel;
402 std::vector<typename Kernel::FT> candidates;
403 candidates.reserve(polygon_boundary.size() - 1);
404
405 // Initialize the range of faces to be checked for intersection: they are
406 // {0,..,n_faces-1}\setminus the current face index face_index.
407 std::vector<unsigned int> face_indices(polygon_boundary.size());
408 std::iota(face_indices.begin(), face_indices.end(), 0); // fill the range
409 face_indices.erase(face_indices.cbegin() +
410 face_index); // remove current index
411
412 if constexpr (dim == 2)
413 {
414 typename Kernel::Segment_2 face_segm(
415 {polygon_boundary[face_index]->vertex(0)[0],
416 polygon_boundary[face_index]->vertex(0)[1]},
417 {polygon_boundary[face_index]->vertex(1)[0],
418 polygon_boundary[face_index]->vertex(1)[1]});
419
420 // Shoot a ray from the midpoint of the face in the orthogonal direction
421 // given by deal.II normals
422 const auto &midpoint = CGAL::midpoint(face_segm);
423 // deal.II normal is always outward, flip the direction
424 const typename Kernel::Vector_2 orthogonal_direction{-deal_normal[0],
425 -deal_normal[1]};
426 const typename Kernel::Ray_2 ray(midpoint, orthogonal_direction);
427 for (const auto f : face_indices)
428 {
429 typename Kernel::Segment_2 segm({polygon_boundary[f]->vertex(0)[0],
430 polygon_boundary[f]->vertex(0)[1]},
431 {polygon_boundary[f]->vertex(1)[0],
432 polygon_boundary[f]->vertex(
433 1)[1]});
434
435 if (CGAL::do_intersect(ray, segm))
436 candidates.push_back(CGAL::squared_distance(midpoint, segm));
437 }
438 return std::sqrt(CGAL::to_double(
439 *std::min_element(candidates.cbegin(), candidates.cend())));
440 }
441 else if constexpr (dim == 3)
442 {
443 const typename Kernel::Point_3 &center{
444 polygon_boundary[face_index]->center()[0],
445 polygon_boundary[face_index]->center()[1],
446 polygon_boundary[face_index]->center()[2]};
447 // deal.II normal is always outward, flip the direction
448 const typename Kernel::Vector_3 orthogonal_direction{-deal_normal[0],
449 -deal_normal[1],
450 -deal_normal[2]};
451 const typename Kernel::Ray_3 ray(center, orthogonal_direction);
452
453 for (const auto f : face_indices)
454 {
455 // split the face into 2 triangles and compute distances
456 typename Kernel::Triangle_3 first_triangle(
457 {polygon_boundary[f]->vertex(0)[0],
458 polygon_boundary[f]->vertex(0)[1],
459 polygon_boundary[f]->vertex(0)[2]},
460 {polygon_boundary[f]->vertex(1)[0],
461 polygon_boundary[f]->vertex(1)[1],
462 polygon_boundary[f]->vertex(1)[2]},
463 {polygon_boundary[f]->vertex(3)[0],
464 polygon_boundary[f]->vertex(3)[1],
465 polygon_boundary[f]->vertex(3)[2]});
466 typename Kernel::Triangle_3 second_triangle(
467 {polygon_boundary[f]->vertex(0)[0],
468 polygon_boundary[f]->vertex(0)[1],
469 polygon_boundary[f]->vertex(0)[2]},
470 {polygon_boundary[f]->vertex(3)[0],
471 polygon_boundary[f]->vertex(3)[1],
472 polygon_boundary[f]->vertex(3)[2]},
473 {polygon_boundary[f]->vertex(2)[0],
474 polygon_boundary[f]->vertex(2)[1],
475 polygon_boundary[f]->vertex(2)[2]});
476
477 // compute point-triangle distance only if the orthogonal ray
478 // hits the triangle
479 if (CGAL::do_intersect(ray, first_triangle))
480 candidates.push_back(
481 CGAL::squared_distance(center, first_triangle));
482 if (CGAL::do_intersect(ray, second_triangle))
483 candidates.push_back(
484 CGAL::squared_distance(center, second_triangle));
485 }
486
487 return std::sqrt(CGAL::to_double(
488 *std::min_element(candidates.cbegin(), candidates.cend())));
489 }
490 else
491 {
492 Assert(false, ExcImpossibleInDim(dim));
493 (void)face_index;
494 (void)polygon_boundary;
495 return {};
496 }
497
498#else
499
500 Assert(false, ExcNeedsCGAL());
501 (void)face_index;
502 (void)polygon_boundary;
503 return {};
504#endif
505 }

515 template <int dim, int spacedim, typename VectorType>
516 void
517 interpolate_to_fine_grid(
518 const AgglomerationHandler<dim, spacedim> &agglomeration_handler,
519 VectorType &dst,
520 const VectorType &src)
521 {
522 Assert((dim == spacedim), ExcNotImplemented());
523 Assert(
524 dst.size() == 0,
525 ExcMessage(
526 "The destination vector must the empt upon calling this function."));
527
528 using NumberType = typename VectorType::value_type;
529 constexpr bool is_trilinos_vector =
530 std::is_same_v<VectorType, TrilinosWrappers::MPI::Vector>;
531 using MatrixType = std::conditional_t<is_trilinos_vector,
532 TrilinosWrappers::SparseMatrix,
533 SparseMatrix<NumberType>>;
534
535 MatrixType interpolation_matrix;
536
537 [[maybe_unused]]
538 typename std::conditional_t<!is_trilinos_vector, SparsityPattern, void *>
539 sp;
540
541 // Get some info from the handler
542 const DoFHandler<dim, spacedim> &agglo_dh = agglomeration_handler.agglo_dh;
543
544 DoFHandler<dim, spacedim> *output_dh =
545 const_cast<DoFHandler<dim, spacedim> *>(&agglomeration_handler.output_dh);
546 const FiniteElement<dim, spacedim> &fe = agglomeration_handler.get_fe();
547 const Triangulation<dim, spacedim> &tria =
548 agglomeration_handler.get_triangulation();
549 const auto &bboxes = agglomeration_handler.get_local_bboxes();
550
551 // Setup an auxiliary DoFHandler for output purposes
552 output_dh->reinit(tria);
553 output_dh->distribute_dofs(fe);
554
555 const IndexSet &locally_owned_dofs = output_dh->locally_owned_dofs();
556 const IndexSet &locally_owned_dofs_agglo = agglo_dh.locally_owned_dofs();
557
558 std::conditional_t<is_trilinos_vector,
559 TrilinosWrappers::SparsityPattern,
560 DynamicSparsityPattern>
561 dsp;
562
563 if constexpr (is_trilinos_vector)
564 dsp.reinit(locally_owned_dofs,
565 locally_owned_dofs_agglo,
566 tria.get_communicator());
567 else
568 dsp.reinit(output_dh->n_dofs(),
569 agglo_dh.n_dofs(),
570 output_dh->locally_owned_dofs());
571
572 std::vector<types::global_dof_index> agglo_dof_indices(fe.dofs_per_cell);
573 std::vector<types::global_dof_index> standard_dof_indices(fe.dofs_per_cell);
574 std::vector<types::global_dof_index> output_dof_indices(fe.dofs_per_cell);
575
576 Quadrature<dim> quad(fe.get_unit_support_points());
577 FEValues<dim, spacedim> output_fe_values(fe,
578 quad,
579 update_quadrature_points);
580
581 for (const auto &cell : agglo_dh.active_cell_iterators())
582 if (cell->is_locally_owned())
583 {
584 if (agglomeration_handler.is_master_cell(cell))
585 {
586 auto slaves = agglomeration_handler.get_slaves_of_idx(
587 cell->active_cell_index());
588 slaves.emplace_back(cell);
589
590 cell->get_dof_indices(agglo_dof_indices);
591
592 for (const auto &slave : slaves)
593 {
594 // addd master-slave relationship
595 const auto slave_output =
596 slave->as_dof_handler_iterator(*output_dh);
597 slave_output->get_dof_indices(output_dof_indices);
598 for (const auto row : output_dof_indices)
599 dsp.add_entries(row,
600 agglo_dof_indices.begin(),
601 agglo_dof_indices.end());
602 }
603 }
604 }
605
606
607 const auto assemble_interpolation_matrix = [&]() {
608 FullMatrix<NumberType> local_matrix(fe.dofs_per_cell, fe.dofs_per_cell);
609 std::vector<Point<dim>> reference_q_points(fe.dofs_per_cell);
610
611 // Dummy AffineConstraints, only needed for loc2glb
612 AffineConstraints<NumberType> c;
613 c.close();
614
615 for (const auto &cell : agglo_dh.active_cell_iterators())
616 if (cell->is_locally_owned())
617 {
618 if (agglomeration_handler.is_master_cell(cell))
619 {
620 auto slaves = agglomeration_handler.get_slaves_of_idx(
621 cell->active_cell_index());
622 slaves.emplace_back(cell);
623
624 cell->get_dof_indices(agglo_dof_indices);
625
626 const types::global_cell_index polytope_index =
627 agglomeration_handler.cell_to_polytope_index(cell);
628
629 // Get the box of this agglomerate.
630 const BoundingBox<dim> &box = bboxes[polytope_index];
631
632 for (const auto &slave : slaves)
633 {
634 // add master-slave relationship
635 const auto slave_output =
636 slave->as_dof_handler_iterator(*output_dh);
637
638 slave_output->get_dof_indices(output_dof_indices);
639 output_fe_values.reinit(slave_output);
640
641 local_matrix = 0.;
642
643 const auto &q_points =
644 output_fe_values.get_quadrature_points();
645 for (const auto i : output_fe_values.dof_indices())
646 {
647 const auto &p = box.real_to_unit(q_points[i]);
648 for (const auto j : output_fe_values.dof_indices())
649 {
650 local_matrix(i, j) = fe.shape_value(j, p);
651 }
652 }
653 c.distribute_local_to_global(local_matrix,
654 output_dof_indices,
655 agglo_dof_indices,
656 interpolation_matrix);
657 }
658 }
659 }
660 };
661
662
663 if constexpr (std::is_same_v<MatrixType, TrilinosWrappers::SparseMatrix>)
664 {
665 dsp.compress();
666 interpolation_matrix.reinit(dsp);
667 dst.reinit(locally_owned_dofs);
668 assemble_interpolation_matrix();
669 }
670 else if constexpr (std::is_same_v<MatrixType, SparseMatrix<NumberType>>)
671 {
672 sp.copy_from(dsp);
673 interpolation_matrix.reinit(sp);
674 dst.reinit(output_dh->n_dofs());
675 assemble_interpolation_matrix();
676 }
677 else
678 {
679 // PETSc, LA::d::v options not implemented.
680 (void)agglomeration_handler;
681 (void)dst;
682 (void)src;
683 AssertThrow(false, ExcNotImplemented());
684 }
685
686 // If tria is distributed
687 if (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
688 &tria) != nullptr)
689 interpolation_matrix.compress(VectorOperation::add);
690
691 // Finally, perform the interpolation.
692 interpolation_matrix.vmult(dst, src);
693 }

732 template <int dim, int spacedim>
733 void
734 partition_locally_owned_regions(const unsigned int n_partitions,
735 Triangulation<dim, spacedim> &triangulation,
736 const SparsityTools::Partitioner partitioner)
737 {
738 AssertDimension(dim, spacedim);
739 Assert(n_partitions > 0,
740 ExcMessage("Invalid number of partitions, you provided " +
741 std::to_string(n_partitions)));
742
743 auto parallel_triangulation =
744 dynamic_cast<parallel::fullydistributed::Triangulation<dim, spacedim> *>(
745 &triangulation);
746 Assert(
747 (parallel_triangulation != nullptr),
748 ExcMessage(
749 "Only fully distributed triangulations are supported. If you are using"
750 "a parallel::distributed::triangulation, you must convert it to a fully"
751 "distributed as explained in the documentation."));
752
753 // check for an easy return
754 if (n_partitions == 1)
755 {
756 for (const auto &cell : parallel_triangulation->active_cell_iterators())
757 if (cell->is_locally_owned())
758 cell->set_material_id(0);
759 return;
760 }
761
762 // collect all locally owned cells
763 std::vector<types::global_cell_index> locally_owned_cells;
764 for (const auto &cell : triangulation.active_cell_iterators())
765 if (cell->is_locally_owned())
766 locally_owned_cells.push_back(cell->active_cell_index());
767
768 DynamicSparsityPattern cell_connectivity;
769 internal::get_face_connectivity_of_cells(*parallel_triangulation,
770 cell_connectivity,
771 locally_owned_cells);
772

774 sp_cell_connectivity.copy_from(cell_connectivity);

798 template <int dim>
799 std::
800 tuple<std::vector<double>, std::vector<double>, std::vector<double>, double>
801 compute_quality_metrics(const AgglomerationHandler<dim> &ah)
802 {
803 static_assert(dim == 2); // only 2D case is implemented.
804#ifdef DEAL_II_WITH_CGAL
805 using Kernel = CGAL::Exact_predicates_exact_constructions_kernel_with_sqrt;
806 using Polygon_with_holes = typename CGAL::Polygon_with_holes_2<Kernel>;
807 using Gt = typename CGAL::Segment_Delaunay_graph_traits_2<Kernel>;
808 using SDG2 = typename CGAL::Segment_Delaunay_graph_2<Gt>;
809 using CDT = typename CGAL::Constrained_Delaunay_triangulation_2<Kernel>;
810 using CDTP = typename CGAL::Constrained_triangulation_plus_2<CDT>;
811 using Point = typename CDTP::Point;
812 using Cid = typename CDTP::Constraint_id;
813 using Vertex_handle = typename CDTP::Vertex_handle;
814
815
816 const auto compute_radius_inscribed_circle =
817 [](const CGAL::Polygon_2<Kernel> &polygon) -> double {
818 SDG2 sdg;
819
820 sdg.insert_segments(polygon.edges_begin(), polygon.edges_end());
821
822 double sd = 0, sqdist = 0;
823 typename SDG2::Finite_faces_iterator fit = sdg.finite_faces_begin();
824 for (; fit != sdg.finite_faces_end(); ++fit)
825 {
826 typename Kernel::Point_2 pp = sdg.primal(fit);
827 for (int i = 0; i < 3; ++i)
828 {
829 assert(!sdg.is_infinite(fit->vertex(i)));
830 if (fit->vertex(i)->site().is_segment())
831 {
832 typename Kernel::Segment_2 s =
833 fit->vertex(i)->site().segment();
834 sqdist = CGAL::to_double(CGAL::squared_distance(pp, s));
835 }
836 else
837 {
838 typename Kernel::Point_2 p = fit->vertex(i)->site().point();
839 sqdist = CGAL::to_double(CGAL::squared_distance(pp, p));
840 }
841 }
842
843 if (polygon.bounded_side(pp) == CGAL::ON_BOUNDED_SIDE)
844 sd = std::max(sqdist, sd);
845 }
846
847 return std::sqrt(sd);
848 };
849
850 const auto mesh_size = [&ah]() -> double {
851 double hmax = 0.;
852 for (const auto &polytope : ah.polytope_iterators())
853 if (polytope->is_locally_owned())
854 {
855 const double diameter = polytope->diameter();
856 if (diameter > hmax)
857 hmax = diameter;
858 }
859 return hmax;
860 }();
861
862
863 // vectors holding quality metrics
864
865 // ration between radius of radius_inscribed_circle and circumscribed circle
866 std::vector<double> circle_ratios;
867 std::vector<double> unformity_factors; // diameter of element over mesh size
868 std::vector<double>
869 box_ratio; // ratio between measure of bbox and measure of element.
870
871 const std::vector<BoundingBox<dim>> &bboxes = ah.get_local_bboxes();
872 // Loop over all polytopes and compute metrics.
873 for (const auto &polytope : ah.polytope_iterators())
874 {
875 if (polytope->is_locally_owned())
876 {
877 const std::vector<typename Triangulation<dim>::active_face_iterator>
878 &boundary = polytope->polytope_boundary();
879
880 const double diameter = polytope->diameter();
881 const double radius_circumscribed_circle = .5 * diameter;
882
883 CDTP cdtp;
884 for (unsigned int f = 0; f < boundary.size(); f += 1)
885 {
886 // polyline
887 cdtp.insert_constraint(
888 {boundary[f]->vertex(0)[0], boundary[f]->vertex(0)[1]},
889 {boundary[f]->vertex(1)[0], boundary[f]->vertex(1)[1]});
890 }
891 cdtp.split_subconstraint_graph_into_constraints();
892
893 CGAL::Polygon_2<Kernel> outer_polygon;
894 auto it = outer_polygon.vertices_begin();
895 for (typename CDTP::Constraint_id cid : cdtp.constraints())
896 {
897 for (typename CDTP::Vertex_handle vh :
898 cdtp.vertices_in_constraint(cid))
899 {
900 it = outer_polygon.insert(outer_polygon.vertices_end(),
901 vh->point());
902 }
903 }
904 outer_polygon.erase(it); // remove duplicate final point
905
906 const double radius_inscribed_circle =
907 compute_radius_inscribed_circle(outer_polygon);
908
909 circle_ratios.push_back(radius_inscribed_circle /
910 radius_circumscribed_circle);
911 unformity_factors.push_back(diameter / mesh_size);
912
913 // box_ratio
914
915 const auto &agglo_values = ah.reinit(polytope);
916 const double measure_element =
917 std::accumulate(agglo_values.get_JxW_values().cbegin(),
918 agglo_values.get_JxW_values().cend(),
919 0.);
920 box_ratio.push_back(measure_element /
921 bboxes[polytope->index()].volume());
922 }
923 }
924
925
926
927 // Get all of the local bounding boxes
928 double covering_bboxes = 0.;
929 for (unsigned int i = 0; i < bboxes.size(); ++i)
930 covering_bboxes += bboxes[i].volume();
931
932 const double overlap_factor =
933 Utilities::MPI::sum(covering_bboxes,
934 ah.get_dof_handler().get_communicator()) /
935 GridTools::volume(ah.get_triangulation()); // assuming a linear mapping
936
937
938
939 return {unformity_factors, circle_ratios, box_ratio, overlap_factor};
940#else
941
942 (void)ah;
943 return {};
944#endif
945 }
