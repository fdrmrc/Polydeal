169 template <typename Value,
170 typename Options,
171 typename Translator,
172 typename Box,
173 typename Allocators>
174 void
175 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
176 const Rtree_visitor::InternalNode &node)
177 {
178 using elements_type =
179 typename boost::geometry::index::detail::rtree::elements_type<
180 InternalNode>::type; // pairs of bounding box and pointer to child node
181 const elements_type &elements =
182 boost::geometry::index::detail::rtree::elements(node);
183
184 if (level < target_level)
185 {
186 size_t level_backup = level;
187 ++level;
188
189 for (typename elements_type::const_iterator it = elements.begin();
190 it != elements.end();
191 ++it)
192 {
193 boost::geometry::index::detail::rtree::apply_visitor(*this,
194 *it->second);
195 }
196
197 level = level_backup;
198 }
199 else if (level == target_level)
200 {
201 // const unsigned int n_children = elements.size();
202 const auto offset = agglomerates.size();
203 agglomerates.resize(offset + 1);
204 row_ptr.resize(row_ptr.size() + 1);
205 next_level_leafs_processed = 0;
206 row_ptr.back().push_back(
207 next_level_leafs_processed); // convention: row_ptr[0]=0
208 size_t level_backup = level;
209
210 ++level;
211 for (const auto &child : elements)
212 {
213 boost::geometry::index::detail::rtree::apply_visitor(*this,
214 *child.second);
215 }
216 // Done with node number 'node_counter'
217
218 ++node_counter; // visited all children of an internal node
219
220 level = level_backup;
221 }
222 else if (level > target_level)
223 {
224 // Keep visiting until you go to the leafs.
225 size_t level_backup = level;
226
227 ++level;
228
229 for (const auto &child : elements)
230 {
231 boost::geometry::index::detail::rtree::apply_visitor(*this,
232 *child.second);
233 }
234 level = level_backup;
235 row_ptr[node_counter].push_back(next_level_leafs_processed);
236 }
237 }

246 void
247 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
248 const Rtree_visitor::Leaf &leaf)
249 {
250 using elements_type =
251 typename boost::geometry::index::detail::rtree::elements_type<
252 Leaf>::type; // pairs of bounding box and pointer to child node
253 const elements_type &elements =
254 boost::geometry::index::detail::rtree::elements(leaf);
255
256
257 for (const auto &it : elements)
258 {
259 agglomerates[node_counter].push_back(it.second);
260 }
261 next_level_leafs_processed += elements.size();
262 }

264 template <typename Rtree>
265 inline std::pair<
266 std::vector<std::vector<unsigned int>>,
267 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
268 typename Rtree::indexable_type>::value>::active_cell_iterator>>>
269 extract_children_of_level(const Rtree &tree, const unsigned int level)
270 {
271 using RtreeView =
272 boost::geometry::index::detail::rtree::utilities::view<Rtree>;
273 RtreeView rtv(tree);
274
275 std::vector<std::vector<unsigned int>> csrs;
276 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
277 typename Rtree::indexable_type>::value>::active_cell_iterator>>
278 agglomerates;
279
280 if (rtv.depth() == 0)
281 {
282 // The below algorithm does not work for `rtv.depth()==0`, which might
283 // happen if the number entries in the tree is too small.
284 // In this case, simply return a single bounding box.
285 agglomerates.resize(1);
286 agglomerates[0].resize(1);
287 csrs.resize(1);
288 csrs[0].resize(1);
289 }
290 else
291 {
292 const unsigned int target_level =
293 std::min<unsigned int>(level, rtv.depth());
294
295 Rtree_visitor<typename RtreeView::value_type,
296 typename RtreeView::options_type,
297 typename RtreeView::translator_type,
298 typename RtreeView::box_type,
299 typename RtreeView::allocators_type>
300 node_visitor(rtv.translator(), target_level, agglomerates, csrs);
301 rtv.apply_visitor(node_visitor);
302 }
303 AssertDimension(agglomerates.size(), csrs.size());
304
305 return {csrs, agglomerates};
306 }

310 template <int dim,
311 typename Kernel = CGAL::Exact_predicates_exact_constructions_kernel,
312 typename Number = double>
313 Number
314 compute_h_orthogonal(
315 const unsigned int face_index,
316 const std::vector<typename Triangulation<dim>::active_face_iterator>
317 & polygon_boundary,
318 const Tensor<1, dim> &deal_normal)
319 {
320#ifdef DEAL_II_WITH_CGAL
321
322 std::vector<typename Kernel::FT> candidates;
323 candidates.reserve(polygon_boundary.size() - 1);
324
325 // Initialize the range of faces to be checked for intersection: they are
326 // {0,..,n_faces-1}\setminus the current face index face_index.
327 std::vector<unsigned int> face_indices(polygon_boundary.size());
328 std::iota(face_indices.begin(), face_indices.end(), 0); // fill the range
329 face_indices.erase(face_indices.cbegin() +
330 face_index); // remove current index
331
332 if constexpr (dim == 2)
333 {
334 typename Kernel::Segment_2 face_segm(
335 {polygon_boundary[face_index]->vertex(0)[0],
336 polygon_boundary[face_index]->vertex(0)[1]},
337 {polygon_boundary[face_index]->vertex(1)[0],
338 polygon_boundary[face_index]->vertex(1)[1]});
339
340 // Shoot a ray from the midpoint of the face in the orthogonal direction
341 // given by deal.II normals
342 const auto &midpoint = CGAL::midpoint(face_segm);
343 // deal.II normal is always outward, flip the direction
344 const typename Kernel::Vector_2 orthogonal_direction{-deal_normal[0],
345 -deal_normal[1]};
346 const typename Kernel::Ray_2 ray(midpoint, orthogonal_direction);
347 for (const auto f : face_indices)
348 {
349 typename Kernel::Segment_2 segm({polygon_boundary[f]->vertex(0)[0],
350 polygon_boundary[f]->vertex(0)[1]},
351 {polygon_boundary[f]->vertex(1)[0],
352 polygon_boundary[f]->vertex(
353 1)[1]});
354
355 if (CGAL::do_intersect(ray, segm))
356 candidates.push_back(CGAL::squared_distance(midpoint, segm));
357 }
358 return std::sqrt(CGAL::to_double(
359 *std::min_element(candidates.cbegin(), candidates.cend())));
360 }
361 else if constexpr (dim == 3)
362 {
363 const typename Kernel::Point_3 &center{
364 polygon_boundary[face_index]->center()[0],
365 polygon_boundary[face_index]->center()[1],
366 polygon_boundary[face_index]->center()[2]};
367 // deal.II normal is always outward, flip the direction
368 const typename Kernel::Vector_3 orthogonal_direction{-deal_normal[0],
369 -deal_normal[1],
370 -deal_normal[2]};
371 const typename Kernel::Ray_3 ray(center, orthogonal_direction);
372
373 for (const auto f : face_indices)
374 {
375 // split the face into 2 triangles and compute distances
376 typename Kernel::Triangle_3 first_triangle(
377 {polygon_boundary[f]->vertex(0)[0],
378 polygon_boundary[f]->vertex(0)[1],
379 polygon_boundary[f]->vertex(0)[2]},
380 {polygon_boundary[f]->vertex(1)[0],
381 polygon_boundary[f]->vertex(1)[1],
382 polygon_boundary[f]->vertex(1)[2]},
383 {polygon_boundary[f]->vertex(3)[0],
384 polygon_boundary[f]->vertex(3)[1],
385 polygon_boundary[f]->vertex(3)[2]});
386 typename Kernel::Triangle_3 second_triangle(
387 {polygon_boundary[f]->vertex(0)[0],
388 polygon_boundary[f]->vertex(0)[1],
389 polygon_boundary[f]->vertex(0)[2]},
390 {polygon_boundary[f]->vertex(3)[0],
391 polygon_boundary[f]->vertex(3)[1],
392 polygon_boundary[f]->vertex(3)[2]},
393 {polygon_boundary[f]->vertex(2)[0],
394 polygon_boundary[f]->vertex(2)[1],
395 polygon_boundary[f]->vertex(2)[2]});
396
397 // compute point-triangle distance only if the orthogonal ray
398 // hits the triangle
399 if (CGAL::do_intersect(ray, first_triangle))
400 candidates.push_back(
401 CGAL::squared_distance(center, first_triangle));
402 if (CGAL::do_intersect(ray, second_triangle))
403 candidates.push_back(
404 CGAL::squared_distance(center, second_triangle));
405 }
406
407 return std::sqrt(CGAL::to_double(
408 *std::min_element(candidates.cbegin(), candidates.cend())));
409 }
410 else
411 {
412 Assert(false, ExcImpossibleInDim(dim));
413 (void)face_index;
414 (void)polygon_boundary;
415 return {};
416 }
417
418#else
419
420 Assert(false, ExcNeedsCGAL());
421 (void)face;
422 (void)polygon_boundary;
423 return {};
424#endif
425 }
