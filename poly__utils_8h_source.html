82 template <int dim, int spacedim>
83 void
84 get_face_connectivity_of_cells(
85 const parallel::fullydistributed::Triangulation<dim, spacedim>
86 & triangulation,
87 DynamicSparsityPattern & cell_connectivity,
88 const std::vector<types::global_cell_index> locally_owned_cells)
89 {
90 cell_connectivity.reinit(triangulation.n_locally_owned_active_cells(),
91 triangulation.n_locally_owned_active_cells());
92
93
94 // loop over all cells and their neighbors to build the sparsity
95 // pattern. note that it's a bit hard to enter all the connections when
96 // a neighbor has children since we would need to find out which of its
97 // children is adjacent to the current cell. this problem can be omitted
98 // if we only do something if the neighbor has no children -- in that
99 // case it is either on the same or a coarser level than we are. in
100 // return, we have to add entries in both directions for both cells
101 for (const auto &cell : triangulation.active_cell_iterators())
102 {
103 if (cell->is_locally_owned())
104 {
105 const unsigned int index = cell->active_cell_index();
106 cell_connectivity.add(get_index(locally_owned_cells, index),
107 get_index(locally_owned_cells, index));
108 for (auto f : cell->face_indices())
109 if ((cell->at_boundary(f) == false) &&
110 (cell->neighbor(f)->has_children() == false) &&
111 cell->neighbor(f)->is_locally_owned())
112 {
113 const unsigned int other_index =
114 cell->neighbor(f)->active_cell_index();
115
116 cell_connectivity.add(get_index(locally_owned_cells, index),
117 get_index(locally_owned_cells,
118 other_index));
119 cell_connectivity.add(get_index(locally_owned_cells,
120 other_index),
121 get_index(locally_owned_cells, index));
122 }
123 }
124 }

246 template <typename Value,
247 typename Options,
248 typename Translator,
249 typename Box,
250 typename Allocators>
251 void
252 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
253 const Rtree_visitor::InternalNode &node)
254 {
255 using elements_type =
256 typename boost::geometry::index::detail::rtree::elements_type<
257 InternalNode>::type; // pairs of bounding box and pointer to child node
258 const elements_type &elements =
259 boost::geometry::index::detail::rtree::elements(node);
260
261 if (level < target_level)
262 {
263 size_t level_backup = level;
264 ++level;
265
266 for (typename elements_type::const_iterator it = elements.begin();
267 it != elements.end();
268 ++it)
269 {
270 boost::geometry::index::detail::rtree::apply_visitor(*this,
271 *it->second);
272 }
273
274 level = level_backup;
275 }
276 else if (level == target_level)
277 {
278 // const unsigned int n_children = elements.size();
279 const auto offset = agglomerates.size();
280 agglomerates.resize(offset + 1);
281 row_ptr.resize(row_ptr.size() + 1);
282 next_level_leafs_processed = 0;
283 row_ptr.back().push_back(
284 next_level_leafs_processed); // convention: row_ptr[0]=0
285 size_t level_backup = level;
286
287 ++level;
288 for (const auto &child : elements)
289 {
290 boost::geometry::index::detail::rtree::apply_visitor(*this,
291 *child.second);
292 }
293 // Done with node number 'node_counter'
294
295 ++node_counter; // visited all children of an internal node
296
297 level = level_backup;
298 }
299 else if (level > target_level)
300 {
301 // Keep visiting until you go to the leafs.
302 size_t level_backup = level;
303
304 ++level;
305
306 for (const auto &child : elements)
307 {
308 boost::geometry::index::detail::rtree::apply_visitor(*this,
309 *child.second);
310 }
311 level = level_backup;
312 row_ptr[node_counter].push_back(next_level_leafs_processed);
313 }
314 }

318 template <typename Value,
319 typename Options,
320 typename Translator,
321 typename Box,
322 typename Allocators>
323 void
324 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
325 const Rtree_visitor::Leaf &leaf)
326 {
327 using elements_type =
328 typename boost::geometry::index::detail::rtree::elements_type<
329 Leaf>::type; // pairs of bounding box and pointer to child node
330 const elements_type &elements =
331 boost::geometry::index::detail::rtree::elements(leaf);
332
333
334 for (const auto &it : elements)
335 {
336 agglomerates[node_counter].push_back(it.second);
337 }
338 next_level_leafs_processed += elements.size();
339 }

341 template <typename Rtree>
342 inline std::pair<
343 std::vector<std::vector<unsigned int>>,
344 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
345 typename Rtree::indexable_type>::value>::active_cell_iterator>>>
346 extract_children_of_level(const Rtree &tree, const unsigned int level)
347 {
348 using RtreeView =
349 boost::geometry::index::detail::rtree::utilities::view<Rtree>;
350 RtreeView rtv(tree);
351
352 std::vector<std::vector<unsigned int>> csrs;
353 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
354 typename Rtree::indexable_type>::value>::active_cell_iterator>>
355 agglomerates;
356
357 if (rtv.depth() == 0)
358 {
359 // The below algorithm does not work for `rtv.depth()==0`, which might
360 // happen if the number entries in the tree is too small.
361 // In this case, simply return a single bounding box.
362 agglomerates.resize(1);
363 agglomerates[0].resize(1);
364 csrs.resize(1);
365 csrs[0].resize(1);
366 }
367 else
368 {
369 const unsigned int target_level =
370 std::min<unsigned int>(level, rtv.depth());
371
372 Rtree_visitor<typename RtreeView::value_type,
373 typename RtreeView::options_type,
374 typename RtreeView::translator_type,
375 typename RtreeView::box_type,
376 typename RtreeView::allocators_type>
377 node_visitor(rtv.translator(), target_level, agglomerates, csrs);
378 rtv.apply_visitor(node_visitor);
379 }
380 AssertDimension(agglomerates.size(), csrs.size());
381
382 return {csrs, agglomerates};
383 }

387 template <int dim,
388 typename Kernel = CGAL::Exact_predicates_exact_constructions_kernel,
389 typename Number = double>
390 Number
391 compute_h_orthogonal(
392 const unsigned int face_index,
393 const std::vector<typename Triangulation<dim>::active_face_iterator>
394 & polygon_boundary,
395 const Tensor<1, dim> &deal_normal)
396 {
397#ifdef DEAL_II_WITH_CGAL
398
399 std::vector<typename Kernel::FT> candidates;
400 candidates.reserve(polygon_boundary.size() - 1);
401
402 // Initialize the range of faces to be checked for intersection: they are
403 // {0,..,n_faces-1}\setminus the current face index face_index.
404 std::vector<unsigned int> face_indices(polygon_boundary.size());
405 std::iota(face_indices.begin(), face_indices.end(), 0); // fill the range
406 face_indices.erase(face_indices.cbegin() +
407 face_index); // remove current index
408
409 if constexpr (dim == 2)
410 {
411 typename Kernel::Segment_2 face_segm(
412 {polygon_boundary[face_index]->vertex(0)[0],
413 polygon_boundary[face_index]->vertex(0)[1]},
414 {polygon_boundary[face_index]->vertex(1)[0],
415 polygon_boundary[face_index]->vertex(1)[1]});
416
417 // Shoot a ray from the midpoint of the face in the orthogonal direction
418 // given by deal.II normals
419 const auto &midpoint = CGAL::midpoint(face_segm);
420 // deal.II normal is always outward, flip the direction
421 const typename Kernel::Vector_2 orthogonal_direction{-deal_normal[0],
422 -deal_normal[1]};
423 const typename Kernel::Ray_2 ray(midpoint, orthogonal_direction);
424 for (const auto f : face_indices)
425 {
426 typename Kernel::Segment_2 segm({polygon_boundary[f]->vertex(0)[0],
427 polygon_boundary[f]->vertex(0)[1]},
428 {polygon_boundary[f]->vertex(1)[0],
429 polygon_boundary[f]->vertex(
430 1)[1]});
431
432 if (CGAL::do_intersect(ray, segm))
433 candidates.push_back(CGAL::squared_distance(midpoint, segm));
434 }
435 return std::sqrt(CGAL::to_double(
436 *std::min_element(candidates.cbegin(), candidates.cend())));
437 }
438 else if constexpr (dim == 3)
439 {
440 const typename Kernel::Point_3 &center{
441 polygon_boundary[face_index]->center()[0],
442 polygon_boundary[face_index]->center()[1],
443 polygon_boundary[face_index]->center()[2]};
444 // deal.II normal is always outward, flip the direction
445 const typename Kernel::Vector_3 orthogonal_direction{-deal_normal[0],
446 -deal_normal[1],
447 -deal_normal[2]};
448 const typename Kernel::Ray_3 ray(center, orthogonal_direction);
449
450 for (const auto f : face_indices)
451 {
452 // split the face into 2 triangles and compute distances
453 typename Kernel::Triangle_3 first_triangle(
454 {polygon_boundary[f]->vertex(0)[0],
455 polygon_boundary[f]->vertex(0)[1],
456 polygon_boundary[f]->vertex(0)[2]},
457 {polygon_boundary[f]->vertex(1)[0],
458 polygon_boundary[f]->vertex(1)[1],
459 polygon_boundary[f]->vertex(1)[2]},
460 {polygon_boundary[f]->vertex(3)[0],
461 polygon_boundary[f]->vertex(3)[1],
462 polygon_boundary[f]->vertex(3)[2]});
463 typename Kernel::Triangle_3 second_triangle(
464 {polygon_boundary[f]->vertex(0)[0],
465 polygon_boundary[f]->vertex(0)[1],
466 polygon_boundary[f]->vertex(0)[2]},
467 {polygon_boundary[f]->vertex(3)[0],
468 polygon_boundary[f]->vertex(3)[1],
469 polygon_boundary[f]->vertex(3)[2]},
470 {polygon_boundary[f]->vertex(2)[0],
471 polygon_boundary[f]->vertex(2)[1],
472 polygon_boundary[f]->vertex(2)[2]});
473
474 // compute point-triangle distance only if the orthogonal ray
475 // hits the triangle
476 if (CGAL::do_intersect(ray, first_triangle))
477 candidates.push_back(
478 CGAL::squared_distance(center, first_triangle));
479 if (CGAL::do_intersect(ray, second_triangle))
480 candidates.push_back(
481 CGAL::squared_distance(center, second_triangle));
482 }
483
484 return std::sqrt(CGAL::to_double(
485 *std::min_element(candidates.cbegin(), candidates.cend())));
486 }
487 else
488 {
489 Assert(false, ExcImpossibleInDim(dim));
490 (void)face_index;
491 (void)polygon_boundary;
492 return {};
493 }
494
495#else
496
497 Assert(false, ExcNeedsCGAL());
498 (void)face;
499 (void)polygon_boundary;
500 return {};
501#endif
502 }

512 template <int dim, int spacedim, typename VectorType>
513 void
514 interpolate_to_fine_grid(
515 const AgglomerationHandler<dim, spacedim> &agglomeration_handler,
516 VectorType & dst,
517 const VectorType & src)
518 {
519 Assert((dim == spacedim), ExcNotImplemented());
520 Assert(
521 dst.size() == 0,
522 ExcMessage(
523 "The destination vector must the empt upon calling this function."));
524
525 using NumberType = typename VectorType::value_type;
526 constexpr bool is_trilinos_vector =
527 std::is_same_v<VectorType, TrilinosWrappers::MPI::Vector>;
528 using MatrixType = std::conditional_t<is_trilinos_vector,
529 TrilinosWrappers::SparseMatrix,
530 SparseMatrix<NumberType>>;
531
532 MatrixType interpolation_matrix;
533
534 [[maybe_unused]]
535 typename std::conditional_t<!is_trilinos_vector, SparsityPattern, void *>
536 sp;
537
538 // Get some info from the handler
539 const DoFHandler<dim, spacedim> &agglo_dh = agglomeration_handler.agglo_dh;
540
541 DoFHandler<dim, spacedim> *output_dh =
542 const_cast<DoFHandler<dim, spacedim> *>(&agglomeration_handler.output_dh);
543 const FiniteElement<dim, spacedim> &fe = agglomeration_handler.get_fe();
544 const Triangulation<dim, spacedim> &tria =
545 agglomeration_handler.get_triangulation();
546 const auto &bboxes = agglomeration_handler.get_local_bboxes();
547
548 // Setup an auxiliary DoFHandler for output purposes
549 output_dh->reinit(tria);
550 output_dh->distribute_dofs(fe);
551
552 const IndexSet &locally_owned_dofs = output_dh->locally_owned_dofs();
553 const IndexSet locally_relevant_dofs =
554 DoFTools::extract_locally_relevant_dofs(*output_dh);
555
556 const IndexSet &locally_owned_dofs_agglo = agglo_dh.locally_owned_dofs();
557
558
559 DynamicSparsityPattern dsp(output_dh->n_dofs(), agglo_dh.n_dofs());
560
561 std::vector<types::global_dof_index> agglo_dof_indices(fe.dofs_per_cell);
562 std::vector<types::global_dof_index> standard_dof_indices(fe.dofs_per_cell);
563 std::vector<types::global_dof_index> output_dof_indices(fe.dofs_per_cell);
564
565 Quadrature<dim> quad(fe.get_unit_support_points());
566 FEValues<dim, spacedim> output_fe_values(fe,
567 quad,
568 update_quadrature_points);
569
570 for (const auto &cell : agglo_dh.active_cell_iterators())
571 if (cell->is_locally_owned())
572 {
573 if (agglomeration_handler.is_master_cell(cell))
574 {
575 auto slaves = agglomeration_handler.get_slaves_of_idx(
576 cell->active_cell_index());
577 slaves.emplace_back(cell);
578
579 cell->get_dof_indices(agglo_dof_indices);
580
581 for (const auto &slave : slaves)
582 {
583 // addd master-slave relationship
584 const auto slave_output =
585 slave->as_dof_handler_iterator(*output_dh);
586 slave_output->get_dof_indices(output_dof_indices);
587 for (const auto row : output_dof_indices)
588 dsp.add_entries(row,
589 agglo_dof_indices.begin(),
590 agglo_dof_indices.end());
591 }
592 }
593 else if (agglomeration_handler.is_standard_cell(cell))
594 {
595 cell->get_dof_indices(agglo_dof_indices);
596 const auto standard_output =
597 cell->as_dof_handler_iterator(*output_dh);
598 standard_output->get_dof_indices(standard_dof_indices);
599 for (const auto row : standard_dof_indices)
600 dsp.add_entries(row,
601 agglo_dof_indices.begin(),
602 agglo_dof_indices.end());
603 }
604 }
605
606
607 const auto assemble_interpolation_matrix = [&]() {
608 FullMatrix<NumberType> local_matrix(fe.dofs_per_cell, fe.dofs_per_cell);
609 std::vector<Point<dim>> reference_q_points(fe.dofs_per_cell);
610
611 // Dummy AffineConstraints, only needed for loc2glb
612 AffineConstraints<NumberType> c;
613 c.close();
614
615 for (const auto &cell : agglo_dh.active_cell_iterators())
616 if (cell->is_locally_owned())
617 {
618 if (agglomeration_handler.is_master_cell(cell))
619 {
620 auto slaves = agglomeration_handler.get_slaves_of_idx(
621 cell->active_cell_index());
622 slaves.emplace_back(cell);
623
624 cell->get_dof_indices(agglo_dof_indices);
625
626 const types::global_cell_index polytope_index =
627 agglomeration_handler.cell_to_polytope_index(cell);
628
629 // Get the box of this agglomerate.
630 const BoundingBox<dim> &box = bboxes[polytope_index];
631
632 for (const auto &slave : slaves)
633 {
634 // add master-slave relationship
635 const auto slave_output =
636 slave->as_dof_handler_iterator(*output_dh);
637
638 slave_output->get_dof_indices(output_dof_indices);
639 output_fe_values.reinit(slave_output);
640
641 local_matrix = 0.;
642
643 const auto &q_points =
644 output_fe_values.get_quadrature_points();
645 for (const auto i : output_fe_values.dof_indices())
646 {
647 const auto &p = box.real_to_unit(q_points[i]);
648 for (const auto j : output_fe_values.dof_indices())
649 {
650 local_matrix(i, j) = fe.shape_value(j, p);
651 }
652 }
653 c.distribute_local_to_global(local_matrix,
654 output_dof_indices,
655 agglo_dof_indices,
656 interpolation_matrix);
657 }
658 }
659 else if (agglomeration_handler.is_standard_cell(cell))
660 {
661 cell->get_dof_indices(agglo_dof_indices);
662
663 const auto standard_output =
664 cell->as_dof_handler_iterator(*output_dh);
665
666 standard_output->get_dof_indices(standard_dof_indices);
667 output_fe_values.reinit(standard_output);
668
669 local_matrix = 0.;
670 for (const auto i : output_fe_values.dof_indices())
671 local_matrix(i, i) = 1.;
672 c.distribute_local_to_global(local_matrix,
673 standard_dof_indices,
674 agglo_dof_indices,
675 interpolation_matrix);
676 }
677 }
678 };
679
680
681 if constexpr (std::is_same_v<MatrixType, TrilinosWrappers::SparseMatrix>)
682 {
683 const MPI_Comm &communicator = tria.get_communicator();
684 SparsityTools::distribute_sparsity_pattern(dsp,
685 locally_owned_dofs,
686 communicator,
687 locally_relevant_dofs);
688
689 interpolation_matrix.reinit(locally_owned_dofs,
690 locally_owned_dofs_agglo,
691 dsp,
692 communicator);
693 dst.reinit(locally_owned_dofs);
694 assemble_interpolation_matrix();
695 }
696 else if constexpr (std::is_same_v<MatrixType, SparseMatrix<NumberType>>)
697 {
698 sp.copy_from(dsp);
699 interpolation_matrix.reinit(sp);
700 dst.reinit(output_dh->n_dofs());
701 assemble_interpolation_matrix();
702 }
703 else
704 {
705 // PETSc, LA::d::v options not implemented.
706 (void)agglomeration_handler;
707 (void)dst;
708 (void)src;
709 AssertThrow(false, ExcNotImplemented());
710 }
711
712 // If tria is distributed
713 if (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
714 &tria) != nullptr)
715 interpolation_matrix.compress(VectorOperation::add);
716
717 // Finally, perform the interpolation.
718 interpolation_matrix.vmult(dst, src);
719 }

726 template <int dim, int spacedim = dim>
727 void
728 collect_cells_for_agglomeration(
729 const Triangulation<dim, spacedim> & tria,
730 const std::vector<types::global_cell_index> &cell_idxs,
731 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
732 &cells_to_be_agglomerated)

741 {

743 }

758 template <int dim, int spacedim>
759 void
760 partition_locally_owned_regions(const unsigned int n_partitions,
761 Triangulation<dim, spacedim> &triangulation,
762 const SparsityTools::Partitioner partitioner)
763 {
764 AssertDimension(dim, spacedim);
765 Assert(n_partitions > 0,
766 ExcMessage("Invalid number of partitions, you provided " +
767 std::to_string(n_partitions)));
768
769 auto parallel_triangulation =
770 dynamic_cast<parallel::fullydistributed::Triangulation<dim, spacedim> *>(
771 &triangulation);
772 Assert(
773 (parallel_triangulation != nullptr),
774 ExcMessage(
775 "Only fully distributed triangulations are supported. If you are using"
776 "a parallel::distributed::triangulation, you must convert it to a fully"

780 if (n_partitions == 1)
781 {
782 for (const auto &cell : parallel_triangulation->active_cell_iterators())
783 if (cell->is_locally_owned())

785 return;
786 }
787

789 std::vector<types::global_cell_index> locally_owned_cells;
790 for (const auto &cell : triangulation.active_cell_iterators())
791 if (cell->is_locally_owned())
792 locally_owned_cells.push_back(cell->active_cell_index());
793

795 internal::get_face_connectivity_of_cells(*parallel_triangulation,
796 cell_connectivity,
797 locally_owned_cells);
798

800 sp_cell_connectivity.copy_from(cell_connectivity);
801
802 // partition each locally owned connection graph and get
803 // back a vector of indices, one per degree
804 // of freedom (which is associated with a
805 // cell)
806 std::vector<unsigned int> partition_indices(
807 parallel_triangulation->n_locally_owned_active_cells());
808 SparsityTools::partition(sp_cell_connectivity,
809 n_partitions,
810 partition_indices,
811 partitioner);
812
813
814 // finally loop over all cells and set the material ids
815 for (const auto &cell : parallel_triangulation->active_cell_iterators())
816 if (cell->is_locally_owned())
817 cell->set_material_id(
818 partition_indices[internal::get_index(locally_owned_cells,
819 cell->active_cell_index())]);
