182 template <typename Value,
183 typename Options,
184 typename Translator,
185 typename Box,
186 typename Allocators>
187 void
188 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
189 const Rtree_visitor::InternalNode &node)
190 {
191 using elements_type =
192 typename boost::geometry::index::detail::rtree::elements_type<
193 InternalNode>::type; // pairs of bounding box and pointer to child node
194 const elements_type &elements =
195 boost::geometry::index::detail::rtree::elements(node);
196
197 if (level < target_level)
198 {
199 size_t level_backup = level;
200 ++level;
201
202 for (typename elements_type::const_iterator it = elements.begin();
203 it != elements.end();
204 ++it)
205 {
206 boost::geometry::index::detail::rtree::apply_visitor(*this,
207 *it->second);
208 }
209
210 level = level_backup;
211 }
212 else if (level == target_level)
213 {
214 // const unsigned int n_children = elements.size();
215 const auto offset = agglomerates.size();
216 agglomerates.resize(offset + 1);
217 row_ptr.resize(row_ptr.size() + 1);
218 next_level_leafs_processed = 0;
219 row_ptr.back().push_back(
220 next_level_leafs_processed); // convention: row_ptr[0]=0
221 size_t level_backup = level;
222
223 ++level;
224 for (const auto &child : elements)
225 {
226 boost::geometry::index::detail::rtree::apply_visitor(*this,
227 *child.second);
228 }
229 // Done with node number 'node_counter'
230
231 ++node_counter; // visited all children of an internal node
232
233 level = level_backup;
234 }
235 else if (level > target_level)
236 {
237 // Keep visiting until you go to the leafs.
238 size_t level_backup = level;
239
240 ++level;
241
242 for (const auto &child : elements)
243 {
244 boost::geometry::index::detail::rtree::apply_visitor(*this,
245 *child.second);
246 }
247 level = level_backup;
248 row_ptr[node_counter].push_back(next_level_leafs_processed);
249 }
250 }

267 boost::geometry::index::detail::rtree::elements(leaf);

277 template <typename Rtree>
278 inline std::pair<
279 std::vector<std::vector<unsigned int>>,
280 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
281 typename Rtree::indexable_type>::value>::active_cell_iterator>>>
282 extract_children_of_level(const Rtree &tree, const unsigned int level)
283 {
284 using RtreeView =
285 boost::geometry::index::detail::rtree::utilities::view<Rtree>;
286 RtreeView rtv(tree);
287
288 std::vector<std::vector<unsigned int>> csrs;
289 std::vector<std::vector<typename Triangulation<boost::geometry::dimension<
290 typename Rtree::indexable_type>::value>::active_cell_iterator>>
291 agglomerates;
292
293 if (rtv.depth() == 0)
294 {
295 // The below algorithm does not work for `rtv.depth()==0`, which might
296 // happen if the number entries in the tree is too small.
297 // In this case, simply return a single bounding box.
298 agglomerates.resize(1);
299 agglomerates[0].resize(1);
300 csrs.resize(1);
301 csrs[0].resize(1);
302 }
303 else
304 {
305 const unsigned int target_level =
306 std::min<unsigned int>(level, rtv.depth());
307
308 Rtree_visitor<typename RtreeView::value_type,
309 typename RtreeView::options_type,
310 typename RtreeView::translator_type,
311 typename RtreeView::box_type,
312 typename RtreeView::allocators_type>
313 node_visitor(rtv.translator(), target_level, agglomerates, csrs);
314 rtv.apply_visitor(node_visitor);
315 }
316 AssertDimension(agglomerates.size(), csrs.size());
317
318 return {csrs, agglomerates};
319 }

323 template <int dim,
324 typename Kernel = CGAL::Exact_predicates_exact_constructions_kernel,
325 typename Number = double>
326 Number
327 compute_h_orthogonal(
328 const unsigned int face_index,
329 const std::vector<typename Triangulation<dim>::active_face_iterator>
330 & polygon_boundary,
331 const Tensor<1, dim> &deal_normal)
332 {
333#ifdef DEAL_II_WITH_CGAL
334
335 std::vector<typename Kernel::FT> candidates;
336 candidates.reserve(polygon_boundary.size() - 1);
337
338 // Initialize the range of faces to be checked for intersection: they are
339 // {0,..,n_faces-1}\setminus the current face index face_index.
340 std::vector<unsigned int> face_indices(polygon_boundary.size());
341 std::iota(face_indices.begin(), face_indices.end(), 0); // fill the range
342 face_indices.erase(face_indices.cbegin() +
343 face_index); // remove current index
344
345 if constexpr (dim == 2)
346 {
347 typename Kernel::Segment_2 face_segm(
348 {polygon_boundary[face_index]->vertex(0)[0],
349 polygon_boundary[face_index]->vertex(0)[1]},
350 {polygon_boundary[face_index]->vertex(1)[0],
351 polygon_boundary[face_index]->vertex(1)[1]});
352
353 // Shoot a ray from the midpoint of the face in the orthogonal direction
354 // given by deal.II normals
355 const auto &midpoint = CGAL::midpoint(face_segm);
356 // deal.II normal is always outward, flip the direction
357 const typename Kernel::Vector_2 orthogonal_direction{-deal_normal[0],
358 -deal_normal[1]};
359 const typename Kernel::Ray_2 ray(midpoint, orthogonal_direction);
360 for (const auto f : face_indices)
361 {
362 typename Kernel::Segment_2 segm({polygon_boundary[f]->vertex(0)[0],
363 polygon_boundary[f]->vertex(0)[1]},
364 {polygon_boundary[f]->vertex(1)[0],
365 polygon_boundary[f]->vertex(
366 1)[1]});
367
368 if (CGAL::do_intersect(ray, segm))
369 candidates.push_back(CGAL::squared_distance(midpoint, segm));
370 }
371 return std::sqrt(CGAL::to_double(
372 *std::min_element(candidates.cbegin(), candidates.cend())));
373 }
374 else if constexpr (dim == 3)
375 {
376 const typename Kernel::Point_3 &center{
377 polygon_boundary[face_index]->center()[0],
378 polygon_boundary[face_index]->center()[1],
379 polygon_boundary[face_index]->center()[2]};
380 // deal.II normal is always outward, flip the direction
381 const typename Kernel::Vector_3 orthogonal_direction{-deal_normal[0],
382 -deal_normal[1],
383 -deal_normal[2]};
384 const typename Kernel::Ray_3 ray(center, orthogonal_direction);
385
386 for (const auto f : face_indices)
387 {
388 // split the face into 2 triangles and compute distances
389 typename Kernel::Triangle_3 first_triangle(
390 {polygon_boundary[f]->vertex(0)[0],
391 polygon_boundary[f]->vertex(0)[1],
392 polygon_boundary[f]->vertex(0)[2]},
393 {polygon_boundary[f]->vertex(1)[0],
394 polygon_boundary[f]->vertex(1)[1],
395 polygon_boundary[f]->vertex(1)[2]},
396 {polygon_boundary[f]->vertex(3)[0],
397 polygon_boundary[f]->vertex(3)[1],
398 polygon_boundary[f]->vertex(3)[2]});
399 typename Kernel::Triangle_3 second_triangle(
400 {polygon_boundary[f]->vertex(0)[0],
401 polygon_boundary[f]->vertex(0)[1],
402 polygon_boundary[f]->vertex(0)[2]},
403 {polygon_boundary[f]->vertex(3)[0],
404 polygon_boundary[f]->vertex(3)[1],
405 polygon_boundary[f]->vertex(3)[2]},
406 {polygon_boundary[f]->vertex(2)[0],
407 polygon_boundary[f]->vertex(2)[1],
408 polygon_boundary[f]->vertex(2)[2]});
409
410 // compute point-triangle distance only if the orthogonal ray
411 // hits the triangle
412 if (CGAL::do_intersect(ray, first_triangle))
413 candidates.push_back(
414 CGAL::squared_distance(center, first_triangle));
415 if (CGAL::do_intersect(ray, second_triangle))
416 candidates.push_back(
417 CGAL::squared_distance(center, second_triangle));
418 }
419
420 return std::sqrt(CGAL::to_double(
421 *std::min_element(candidates.cbegin(), candidates.cend())));
422 }
423 else
424 {
425 Assert(false, ExcImpossibleInDim(dim));
426 (void)face_index;
427 (void)polygon_boundary;
428 return {};
429 }
430
431#else
432
433 Assert(false, ExcNeedsCGAL());
434 (void)face;
435 (void)polygon_boundary;
436 return {};
437#endif
438 }

448 template <int dim, int spacedim, typename VectorType>
449 void
450 interpolate_to_fine_grid(
451 const AgglomerationHandler<dim, spacedim> &agglomeration_handler,
452 VectorType & dst,
453 const VectorType & src)
454 {
455 Assert((dim == spacedim), ExcNotImplemented());
456 Assert(
457 dst.size() == 0,
458 ExcMessage(
459 "The destination vector must the empt upon calling this function."));
460
461 using NumberType = typename VectorType::value_type;
462 using MatrixType = std::conditional_t<
463 std::is_same_v<VectorType, TrilinosWrappers::MPI::Vector>,
464 TrilinosWrappers::SparseMatrix,
465 SparseMatrix<NumberType>>;
466
467 MatrixType interpolation_matrix;
468
469 std::conditional_t<
470 !std::is_same_v<VectorType, TrilinosWrappers::MPI::Vector>,
471 SparsityPattern,
472 void *>
473 sp;
474
475 // Get some info from the handler
476 const DoFHandler<dim, spacedim> &agglo_dh = agglomeration_handler.agglo_dh;
477
478 DoFHandler<dim, spacedim> *output_dh =
479 const_cast<DoFHandler<dim, spacedim> *>(&agglomeration_handler.output_dh);
480 const FiniteElement<dim, spacedim> &fe = agglomeration_handler.get_fe();
481 const Triangulation<dim, spacedim> &tria =
482 agglomeration_handler.get_triangulation();
483 const auto &bboxes = agglomeration_handler.get_local_bboxes();
484
485 // Setup an auxiliary DoFHandler for output purposes
486 output_dh->reinit(tria);
487 output_dh->distribute_dofs(fe);
488
489 const IndexSet &locally_owned_dofs = output_dh->locally_owned_dofs();
490 const IndexSet locally_relevant_dofs =
491 DoFTools::extract_locally_relevant_dofs(*output_dh);
492
493 const IndexSet &locally_owned_dofs_agglo = agglo_dh.locally_owned_dofs();
494
495
496 DynamicSparsityPattern dsp(output_dh->n_dofs(), agglo_dh.n_dofs());
497
498 std::vector<types::global_dof_index> agglo_dof_indices(fe.dofs_per_cell);
499 std::vector<types::global_dof_index> standard_dof_indices(fe.dofs_per_cell);
500 std::vector<types::global_dof_index> output_dof_indices(fe.dofs_per_cell);
501
502 Quadrature<dim> quad(fe.get_unit_support_points());
503 FEValues<dim, spacedim> output_fe_values(fe,
504 quad,
505 update_quadrature_points);
506
507 for (const auto &cell : agglo_dh.active_cell_iterators())
508 if (cell->is_locally_owned())
509 {
510 if (agglomeration_handler.is_master_cell(cell))
511 {
512 auto slaves = agglomeration_handler.get_slaves_of_idx(
513 cell->active_cell_index());
514 slaves.emplace_back(cell);
515
516 cell->get_dof_indices(agglo_dof_indices);
517
518 for (const auto &slave : slaves)
519 {
520 // addd master-slave relationship
521 const auto slave_output =
522 slave->as_dof_handler_iterator(*output_dh);
523 slave_output->get_dof_indices(output_dof_indices);
524 for (const auto row : output_dof_indices)
525 dsp.add_entries(row,
526 agglo_dof_indices.begin(),
527 agglo_dof_indices.end());
528 }
529 }
530 else if (agglomeration_handler.is_standard_cell(cell))
531 {
532 cell->get_dof_indices(agglo_dof_indices);
533 const auto standard_output =
534 cell->as_dof_handler_iterator(*output_dh);
535 standard_output->get_dof_indices(standard_dof_indices);
536 for (const auto row : standard_dof_indices)
537 dsp.add_entries(row,
538 agglo_dof_indices.begin(),
539 agglo_dof_indices.end());
540 }
541 }
542
543
544 const auto assemble_interpolation_matrix = [&]() {
545 FullMatrix<NumberType> local_matrix(fe.dofs_per_cell, fe.dofs_per_cell);
546 std::vector<Point<dim>> reference_q_points(fe.dofs_per_cell);
547
548 // Dummy AffineConstraints, only needed for loc2glb
549 AffineConstraints<NumberType> c;
550 c.close();
551
552 for (const auto &cell : agglo_dh.active_cell_iterators())
553 if (cell->is_locally_owned())
554 {
555 if (agglomeration_handler.is_master_cell(cell))
556 {
557 auto slaves = agglomeration_handler.get_slaves_of_idx(
558 cell->active_cell_index());
559 slaves.emplace_back(cell);
560
561 cell->get_dof_indices(agglo_dof_indices);
562
563 const types::global_cell_index polytope_index =
564 agglomeration_handler.cell_to_polytope_index(cell);
565
566 // Get the box of this agglomerate.
567 const BoundingBox<dim> &box = bboxes[polytope_index];
568
569 for (const auto &slave : slaves)
570 {
571 // add master-slave relationship
572 const auto slave_output =
573 slave->as_dof_handler_iterator(*output_dh);
574
575 slave_output->get_dof_indices(output_dof_indices);
576 output_fe_values.reinit(slave_output);
577
578 local_matrix = 0.;
579
580 const auto &q_points =
581 output_fe_values.get_quadrature_points();
582 for (const auto i : output_fe_values.dof_indices())
583 {
584 const auto &p = box.real_to_unit(q_points[i]);
585 for (const auto j : output_fe_values.dof_indices())
586 {
587 local_matrix(i, j) = fe.shape_value(j, p);
588 }
589 }
590 c.distribute_local_to_global(local_matrix,
591 output_dof_indices,
592 agglo_dof_indices,
593 interpolation_matrix);
594 }
595 }
596 else if (agglomeration_handler.is_standard_cell(cell))
597 {
598 cell->get_dof_indices(agglo_dof_indices);
599
600 const auto standard_output =
601 cell->as_dof_handler_iterator(*output_dh);
602
603 standard_output->get_dof_indices(standard_dof_indices);
604 output_fe_values.reinit(standard_output);
605
606 local_matrix = 0.;
607 for (const auto i : output_fe_values.dof_indices())
608 local_matrix(i, i) = 1.;
609 c.distribute_local_to_global(local_matrix,
610 standard_dof_indices,
611 agglo_dof_indices,
612 interpolation_matrix);
613 }
614 }
615 };
616
617
618 if constexpr (std::is_same_v<MatrixType, TrilinosWrappers::SparseMatrix>)
619 {
620 const MPI_Comm &communicator = tria.get_communicator();
621 SparsityTools::distribute_sparsity_pattern(dsp,
622 locally_owned_dofs,
623 communicator,
624 locally_relevant_dofs);
625
626 interpolation_matrix.reinit(locally_owned_dofs,
627 locally_owned_dofs_agglo,
628 dsp,
629 communicator);
630 dst.reinit(locally_owned_dofs_agglo);
631 assemble_interpolation_matrix();
632 }
633 else if constexpr (std::is_same_v<MatrixType, SparseMatrix<NumberType>>)
634 {
635 sp.copy_from(dsp);
636 interpolation_matrix.reinit(sp);
637 dst.reinit(output_dh->n_dofs());
638 assemble_interpolation_matrix();
639 }
640 else
641 {
642 // PETSc, LA::d::v options not implemented.
643 (void)agglomeration_handler;
644 (void)dst;
645 (void)src;
646 AssertThrow(false, ExcNotImplemented());
647 }
648
649 // If tria is distributed
650 if (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
651 &tria) != nullptr)
652 interpolation_matrix.compress(VectorOperation::add);
653
654 // Finally, perform the interpolation.
655 interpolation_matrix.vmult(dst, src);
656 }
