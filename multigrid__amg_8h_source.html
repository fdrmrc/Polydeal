74 MatrixFreeProjector(
75 const OperatorType &mf_operator_,
76 const std::vector<TrilinosWrappers::SparseMatrix *> transfers_)
77 {
78 // Only DGQ discretizations are supported.
79 Assert(dynamic_cast<const FE_DGQ<dim> *>(
80 &mf_operator_.get_matrix_free()->get_dof_handler().get_fe()) !=
81 nullptr,
82 ExcNotImplemented());
83 using VectorType = LinearAlgebra::distributed::Vector<Number>;
84
85 // Check parallel layout is identical on every level
86 for (unsigned int l = 0; l < transfers_.size(); ++l)
87 Assert((mf_operator_.get_matrix_free()->get_locally_owned_set() ==
88 transfers_[l]->locally_owned_range_indices()),
89 ExcInternalError());
90
91 transfer_matrices.resize(transfers_.size());
92 level_operators.resize(transfers_.size());
93 // get communicator from first Trilinos matrix
94 communicator = transfers_[0]->get_mpi_communicator();
95
96 for (unsigned int l = 0; l < transfers_.size(); ++l)
97 {
98 // Set the pointer to the correct matrix
99 transfer_matrices[l] = transfers_[l];
100
101 // Define vmult-type lambdas for each linear operator.
102 level_operators[l].vmult = [this, l](VectorType &dst,
103 const VectorType &src) {
104 transfer_matrices[l]->vmult(dst, src);
105 };
106 level_operators[l].vmult_add = [this, l](VectorType &dst,
107 const VectorType &src) {
108 transfer_matrices[l]->vmult_add(dst, src);
109 };
110 level_operators[l].Tvmult = [this, l](VectorType &dst,
111 const VectorType &src) {
112 transfer_matrices[l]->Tvmult(dst, src);
113 };
114 level_operators[l].Tvmult_add = [this, l](VectorType &dst,
115 const VectorType &src) {
116 transfer_matrices[l]->Tvmult_add(dst, src);
117 };
118
119 // Inform each linear operator about the parallel layout. Use the
120 // given trilinos matrices.
121 level_operators[l].reinit_domain_vector = [this, l](VectorType &v,
122 bool) {
123 v.reinit(transfer_matrices[l]->locally_owned_domain_indices(),
124 communicator);
125 };
126
127 level_operators[l].reinit_range_vector = [this, l](VectorType &v,
128 bool) {
129 v.reinit(transfer_matrices[l]->locally_owned_range_indices(),
130 communicator);
131 };
132 }
133
134 // Do the same for the matrix-free object.
135 // First, set the pointer
136 mf_operator = &mf_operator_;
137
138 // Then, populate the corresponding lambdas (std::functions)
139 mf_linear_operator.vmult = [this](VectorType &dst,
140 const VectorType &src) {
141 mf_operator->vmult(dst, src);
142 };
143 mf_linear_operator.vmult_add = [this](VectorType &dst,
144 const VectorType &src) {
145 mf_operator->vmult_add(dst, src);
146 };
147 mf_linear_operator.Tvmult = [this](VectorType &dst,
148 const VectorType &src) {
149 mf_operator->Tvmult(dst, src);
150 };
151 mf_linear_operator.Tvmult_add = [this](VectorType &dst,
152 const VectorType &src) {
153 mf_operator->Tvmult_add(dst, src);
154 };
155
156 mf_linear_operator.reinit_domain_vector = [&](VectorType &v, bool) {};
157 mf_linear_operator.reinit_range_vector = [&](VectorType &v, bool) {
158 v.reinit(mf_operator->get_matrix_free()
159 ->get_dof_handler()
160 .locally_owned_dofs(),
161 communicator);
162 };
163 }

173 void
174 compute_level_matrices(
175 MGLevelObject<LinearOperatorMG<VectorType, VectorType>> &mg_matrices)
176 {
177 Assert(mg_matrices.n_levels() > 1,
178 ExcMessage("Vector of matrices set to invalid size."));
179 Assert(!mf_linear_operator.is_null_operator, ExcInternalError());
180
181 using VectorType = LinearAlgebra::distributed::Vector<Number>;
182 const unsigned int min_level = mg_matrices.min_level();
183 const unsigned int max_level = mg_matrices.max_level();
184
185 mg_matrices[min_level] = mf_linear_operator; // finest level
186
187 // do the same, but using transfers to define level matrices
188 for (unsigned int l = min_level + 1; l < max_level; l++)
189 mg_matrices[l] = transpose_operator(level_operators[l - 1]) *
190 mf_linear_operator * level_operators[l - 1];
191 }
192 };

272 void
273 compute_level_matrices(
274 MGLevelObject<std::unique_ptr<MatrixType>> &mg_matrices)
275 {
276 Assert(mg_matrices.n_levels() > 1,
277 ExcMessage("Vector of matrices set to invalid size."));
278 using VectorType = LinearAlgebra::distributed::Vector<Number>;
279
280 const unsigned int min_level = mg_matrices.min_level();
281 const unsigned int max_level = mg_matrices.max_level();
282
283 // mg_matrices[max_level].copy_from(fine_operator); // finest level
284 // MPI_Barrier(communicator);
285 // std::cout << "copied finest operator " << max_level << std::endl;
286
287 // do the same, but using transfers to define level matrices
288 // std::cout << "min level = " << min_level << std::endl;
289 for (unsigned int l = max_level; l-- > min_level;)
290 {
291 // Set parallel layout of intermediate operators AP
292 MatrixType level_operator(
293 transfer_matrices[l]->locally_owned_range_indices(),
294 transfer_matrices[l]->locally_owned_domain_indices(),
295 communicator);
296
297 mg_matrices[l] = std::make_unique<MatrixType>();
298
299 // First, compute AP
300 mg_matrices[l + 1]->mmult(
301 level_operator,
302 *transfer_matrices[l]); // result stored in level_operators[l]
303 // Multiply by the transpose
304 transfer_matrices[l]->Tmmult(*mg_matrices[l], level_operator);
305 }

318 void
319 compute_level_matrices_as_linear_operators(
320 MGLevelObject<std::unique_ptr<MatrixType>> &mg_matrices,
321 MGLevelObject<
322 LinearOperatorMG<LinearAlgebra::distributed::Vector<Number>,
323 LinearAlgebra::distributed::Vector<Number>>>
324 &multigrid_matrices_lo)
325 {
326 Assert(mg_matrices.n_levels() > 1,
327 ExcMessage("Vector of matrices set to invalid size."));
328 using VectorType = LinearAlgebra::distributed::Vector<Number>;
329
330 const unsigned int min_level = mg_matrices.min_level();
331 const unsigned int max_level = mg_matrices.max_level();
332
333 for (unsigned int l = max_level; l-- > min_level;)
334 {
335 // Set parallel layout of intermediate operators AP
336 MatrixType level_operator(
337 transfer_matrices[l]->locally_owned_range_indices(),
338 transfer_matrices[l]->locally_owned_domain_indices(),
339 communicator);
340
341 mg_matrices[l] = std::make_unique<MatrixType>();
342
343 // First, compute AP
344 mg_matrices[l + 1]->mmult(
345 level_operator,
346 *transfer_matrices[l]); // result stored in level_operators[l]
347 // Multiply by the transpose
348 transfer_matrices[l]->Tmmult(*mg_matrices[l], level_operator);
349
350 // Wrap every matrix into a linear operator now.
351 multigrid_matrices_lo[l] =
352 linear_operator_mg<VectorType, VectorType>(*mg_matrices[l]);
353 multigrid_matrices_lo[l].vmult =
354 [&mg_matrices, l](VectorType &dst, const VectorType &src) {
355 mg_matrices[l]->vmult(dst, src);
356 };
357 multigrid_matrices_lo[l].n_rows = mg_matrices[l]->m();
358 multigrid_matrices_lo[l].n_cols = mg_matrices[l]->n();
359 }
360 }
