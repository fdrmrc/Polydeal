163 template <typename Value,
164 typename Options,
165 typename Translator,
166 typename Box,
167 typename Allocators>
168 void
169 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
170 const Rtree_visitor::InternalNode &node)
171 {
172 using elements_type =
173 typename boost::geometry::index::detail::rtree::elements_type<
174 InternalNode>::type; // pairs of bounding box and pointer to child
175 // node
176 const elements_type &elements =
177 boost::geometry::index::detail::rtree::elements(node);
178
179 if (level < target_level)
180 {
181 size_t level_backup = level;
182 ++level;
183
184 for (typename elements_type::const_iterator it = elements.begin();
185 it != elements.end();
186 ++it)
187 {
188 boost::geometry::index::detail::rtree::apply_visitor(*this,
189 *it->second);
190 }
191
192 level = level_backup;
193 }
194 else if (level == target_level)
195 {
196 const auto offset = agglomerates.size();
197 agglomerates.resize(offset + 1);
198 size_t level_backup = level;
199
200 ++level;
201 for (const auto &entry : elements)
202 {
203 boost::geometry::index::detail::rtree::apply_visitor(
204 *this, *entry.second);
205 }
206 // Done with node number 'node_counter' on level target_level.
207
208 ++node_counter; // visited all children of an internal node
209 n_nodes_per_level[target_level]++;
210
211 level = level_backup;
212 }
213 else if (level > target_level)
214 {
215 // I am on a child (internal) node on a deeper level.
216
217 // Keep visiting until you go to the leafs.
218 size_t level_backup = level;
219
220 ++level;
221
222 // looping through entries of node
223 for (const auto &entry : elements)
224 {
225 boost::geometry::index::detail::rtree::apply_visitor(
226 *this, *entry.second);
227 }
228 // done with node on level l > target_level (not just
229 // "target_level+1).
230 n_nodes_per_level[level_backup]++;
231 const types::global_cell_index node_idx =
232 n_nodes_per_level[level_backup] - 1; // so to start from 0
233
234 parent_node_to_children_nodes[{n_nodes_per_level[level_backup - 1],
235 level_backup - 1}]
236 .push_back(node_idx);
237
238 level = level_backup;
239 }
240 }

244 template <typename Value,
245 typename Options,
246 typename Translator,
247 typename Box,
248 typename Allocators>
249 void
250 Rtree_visitor<Value, Options, Translator, Box, Allocators>::operator()(
251 const Rtree_visitor::Leaf &leaf)
252 {
253 using elements_type =
254 typename boost::geometry::index::detail::rtree::elements_type<
255 Leaf>::type; // pairs of bounding box and pointer to child node
256 const elements_type &elements =
257 boost::geometry::index::detail::rtree::elements(leaf);
258
259 for (const auto &it : elements)
260 agglomerates[node_counter].push_back(it.second);
261 }

358 template <int dim, typename RtreeType>
359 const std::vector<
360 std::vector<typename Triangulation<dim>::active_cell_iterator>> &
361 CellsAgglomerator<dim, RtreeType>::extract_agglomerates()
362 {
363 using RtreeView =
364 boost::geometry::index::detail::rtree::utilities::view<RtreeType>;
365 RtreeView rtv(*rtree);
366
367 n_nodes_per_level.resize(
368 rtv.depth()); // store how many nodes we have for each level.
369
370 if (rtv.depth() == 0)
371 {
372 // The below algorithm does not work for `rtv.depth()==0`, which might
373 // happen if the number entries in the tree is too small.
374 agglomerates_on_level.resize(1);
375 agglomerates_on_level[0].resize(1);
376 }
377 else
378 {
379 const unsigned int target_level =
380 std::min<unsigned int>(extraction_level, rtv.depth());
381
382 internal::Rtree_visitor<typename RtreeView::value_type,
383 typename RtreeView::options_type,
384 typename RtreeView::translator_type,
385 typename RtreeView::box_type,
386 typename RtreeView::allocators_type>
387 extractor_visitor(rtv.translator(),
388 target_level,
389 agglomerates_on_level,
390 n_nodes_per_level,
391 parent_node_to_children_nodes);
392
393
394 rtv.apply_visitor(extractor_visitor);
395 }
396 return agglomerates_on_level;
397 }
