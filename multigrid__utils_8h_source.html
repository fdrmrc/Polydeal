216 template <typename VectorType,
217 typename SolverType,
218 typename PreconditionerType,
219 std::enable_if_t<
220 std::is_same_v<VectorType, typename SolverType::vector_type>,
221 VectorType> * = nullptr>
222 void
223 solve(SolverType & solver,
224 const LinearOperator<VectorType> &linear_op,
225 const PreconditionerType & preconditioner,
226 VectorType & dst,
227 const VectorType & src)
228 {
229 const auto &A_inv = inverse_operator(linear_op, solver, preconditioner);
230 dst = A_inv * src;
231 }
232
233 template <typename VectorType,
234 typename SolverType,
235 typename PreconditionerType,
236 std::enable_if_t<
237 !std::is_same_v<VectorType, typename SolverType::vector_type>,
238 VectorType> * = nullptr>
239 void
240 solve(SolverType & solver,
241 const LinearOperator<VectorType> &linear_op,
242 const PreconditionerType & preconditioner,
243 VectorType & dst,
244 const VectorType & src)
245 {
246 typename SolverType::vector_type src_;
247 typename SolverType::vector_type dst_;
248
249 src_ = src;
250 dst_ = dst;
251
252 const auto &A_inv = inverse_operator(linear_op, solver, preconditioner);
253 dst_ = A_inv * src_;
254
255 dst = dst_;
256 }
257 } // namespace MGCoarseGridIterativeSolverLO

410 template <int dim, typename VectorType>
411 void
412 MGTwoLevelTransferAgglomeration<dim, VectorType>::reinit(
413 const AgglomerationHandler<dim> &agglo_handler_fine_,
414 const AgglomerationHandler<dim> &agglo_handler_coarse_)
415 {
416 agglo_handler_fine = &agglo_handler_fine_;
417 agglo_handler_coarse = &agglo_handler_coarse_;
418
419 const DoFHandler<dim> &agglo_dh_fine = agglo_handler_fine->agglo_dh;
420 const DoFHandler<dim> &agglo_dh_coarse = agglo_handler_coarse->agglo_dh;
421
422 Assert((agglo_dh_fine.n_dofs() > agglo_dh_coarse.n_dofs()),
423 ExcMessage(
424 "Coarse DoFHandler has more DoFs than finer DoFHandler."));
425
426 // Setup sparsity pattern
427 DynamicSparsityPattern dsp(agglo_dh_fine.n_dofs(),
428 agglo_dh_coarse.n_dofs());
429
430 const auto & fe = agglo_dh_coarse.get_fe();
431 const unsigned int dofs_per_cell = fe.dofs_per_cell;
432
433 for (std::size_t i = 0; i < agglomerates.size(); ++i)
434 {
435 // coarse agglomerate
436 const auto & agglo = agglomerates[i];
437 std::vector<types::global_cell_index> cell_indices;
438 for (const auto &cell : agglo)
439 cell_indices.push_back(cell->active_cell_index());
440
441 const auto &dof_cell_coarse =
442 agglo[std::distance(std::begin(cell_indices),
443 std::min_element(std::begin(cell_indices),
444 std::end(cell_indices)))]
445 ->as_dof_handler_iterator(
446 agglo_dh_coarse); // min index is master cell
447
448 std::vector<types::global_dof_index> dof_indices_coarse(dofs_per_cell);
449 dof_cell_coarse->get_dof_indices(dof_indices_coarse);
450
451 const auto &crs = crss[i]; // vector of indices for current agglomerate
452
453 for (unsigned int k = 0; k < crs.size() - 1; ++k)
454 {
455 std::vector<types::global_dof_index> dof_indices_fine(
456 dofs_per_cell);
457
458
459 // collect indices of one sub-agglomerate
460 std::vector<types::global_cell_index> cell_indices_fine;
461 for (unsigned int j = crs[k]; j < crs[k + 1]; ++j)
462 cell_indices_fine.push_back(agglo[j]->active_cell_index());
463
464 const unsigned int master_idx =
465 std::distance(std::begin(cell_indices_fine),
466 std::min_element(std::begin(cell_indices_fine),
467 std::end(cell_indices_fine)));
468
469 const auto &dof_cell_fine =
470 agglo[crs[k] + master_idx]->as_dof_handler_iterator(
471 agglo_dh_fine);
472
473 dof_cell_fine->get_dof_indices(dof_indices_fine);
474
475
476 for (const auto row : dof_indices_fine)
477 dsp.add_entries(row,
478 dof_indices_coarse.begin(),
479 dof_indices_coarse.end());
480 }
481 }
482
483 prolongation_sparsity->copy_from(dsp);
484 prolongation_matrix->reinit(*prolongation_sparsity);
485
486
487
488 // Fill prolongation_matrix
489 AffineConstraints<typename VectorType::value_type>
490 c; // dummy constraint, needed only for loc2glb operations.
491 c.close();
492 FullMatrix<typename VectorType::value_type> local_matrix(dofs_per_cell,
493 dofs_per_cell);
494 const std::vector<Point<dim>> & unit_support_points =
495 fe.get_unit_support_points();
496
497 const std::vector<BoundingBox<dim>> bboxes =
498 agglo_handler_coarse->get_local_bboxes();
499
500 for (unsigned int i = 0; i < agglomerates.size(); ++i)
501 {
502 // coarse agglomerate
503 const auto & agglo = agglomerates[i];
504 std::vector<types::global_cell_index> cell_indices;
505 for (const auto &cell : agglo)
506 cell_indices.push_back(cell->active_cell_index());
507
508 const auto &master_cell_coarse =
509 agglo[std::distance(std::begin(cell_indices),
510 std::min_element(std::begin(cell_indices),
511 std::end(cell_indices)))];
512 const auto dof_cell_coarse =
513 master_cell_coarse->as_dof_handler_iterator(
514 agglo_dh_coarse); // min index is master cell
515
516 std::vector<types::global_dof_index> dof_indices_coarse(dofs_per_cell);
517 dof_cell_coarse->get_dof_indices(dof_indices_coarse);
518
519
520
521 const auto &crs = crss[i]; // vector of indices for current agglomerate
522
523 for (unsigned int k = 0; k < crs.size() - 1; ++k)
524 {
525 local_matrix = 0.;
526 std::vector<types::global_dof_index> dof_indices_fine(
527 dofs_per_cell);
528
529 // collect indices of one sub-agglomerate
530 std::vector<types::global_cell_index> cell_indices_fine;
531 for (unsigned int j = crs[k]; j < crs[k + 1]; ++j)
532 cell_indices_fine.push_back(agglo[j]->active_cell_index());
533
534
535 const unsigned int master_idx =
536 std::distance(std::begin(cell_indices_fine),
537 std::min_element(std::begin(cell_indices_fine),
538 std::end(cell_indices_fine)));
539
540 const auto &master_cell_fine = agglo[crs[k] + master_idx];
541
542 const auto &dof_cell_fine =
543 master_cell_fine->as_dof_handler_iterator(
544 agglo_dh_fine); // min index is master cell
545
546 dof_cell_fine->get_dof_indices(dof_indices_fine);
547
548 // compute real location of support points
549 std::vector<Point<dim>> real_qpoints;
550 for (const Point<dim> &p : unit_support_points)
551 real_qpoints.push_back(
552 agglo_handler_fine->euler_mapping->transform_unit_to_real_cell(
553 dof_cell_fine, p));
554 // real_qpoints.push_back(box_fine.unit_to_real(p));
555
556 for (unsigned int i = 0; i < dof_indices_coarse.size(); ++i)
557 {
558 const auto &p =
559 agglo_handler_coarse->euler_mapping
560 ->transform_real_to_unit_cell(dof_cell_coarse,
561 real_qpoints[i]);
562 for (unsigned int j = 0; j < dof_indices_fine.size(); ++j)
563 {
564 local_matrix(i, j) = fe.shape_value(j, p);
565 }
566 }
567
568 c.distribute_local_to_global(local_matrix,
569 dof_indices_fine,
570 dof_indices_coarse,
571 *prolongation_matrix);
572 }
573 }
574 }

694 template <int dim, typename VectorType>
695 template <typename MGTwoLevelTransferObject>
696 MGTransferAgglomeration<dim, VectorType>::MGTransferAgglomeration(
697 const MGLevelObject<MGTwoLevelTransferObject> &transfer,
698 const std::vector<const DoFHandler<dim> *> & dof_handlers_)
699 {
700 const unsigned int min_level = transfer.min_level();
701 const unsigned int max_level = transfer.max_level();
702
703 Assert((dof_handlers_.size() == max_level - min_level + 1),
704 ExcMessage("Wrong number of DoFHanders"));
705 dof_handlers = dof_handlers_;
706
707 this->transfer.resize(min_level, max_level);
708
709 for (unsigned int l = min_level; l <= max_level; ++l)
710 this->transfer[l] =
711 &const_cast<MGTwoLevelTransferAgglomeration<dim, VectorType> &>(
712 static_cast<const MGTwoLevelTransferAgglomeration<dim, VectorType> &>(
713 Utilities::get_underlying_value(transfer[l])));
714 }
