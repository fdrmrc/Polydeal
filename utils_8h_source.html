71 template <int dim, int spacedim, typename MatrixType>
72 void
73 fill_injection_matrix(const AgglomerationHandler<dim, spacedim> &coarse_ah,
74 const AgglomerationHandler<dim, spacedim> &fine_ah,
75 SparsityPattern &sp,
76 MatrixType &matrix)
77 {
78 using NumberType = typename MatrixType::value_type;
79 // First, check that we support the matrix types
80 static constexpr bool is_trilinos_matrix =
81 std::is_same_v<TrilinosWrappers::SparseMatrix, MatrixType>;
82 static constexpr bool is_serial_matrix =
83 std::is_same_v<SparseMatrix<NumberType>, MatrixType>;
84 static constexpr bool is_supported_matrix =
85 is_trilinos_matrix || is_serial_matrix;
86 static_assert(is_supported_matrix);
87 if constexpr (is_trilinos_matrix)
88 Assert(matrix.m() == 0, ExcInternalError());
89 if constexpr (is_serial_matrix)
90 Assert(sp.empty() && matrix.empty(),
91 ExcMessage(
92 "The destination matrix and its sparsity pattern must the empty "
93 "upon calling this function."));
94
95 Assert(coarse_ah.n_dofs() < fine_ah.n_dofs(), ExcInternalError());
96 AssertDimension(dim, spacedim);
97
98 // Get information from the handlers
99 const DoFHandler<dim, spacedim> &coarse_agglo_dh = coarse_ah.agglo_dh;
100 const DoFHandler<dim, spacedim> &fine_agglo_dh = fine_ah.agglo_dh;
101
102 const FiniteElement<dim, spacedim> &fe = coarse_ah.get_fe();
103 const Triangulation<dim, spacedim> &tria = coarse_ah.get_triangulation();
104 const auto &fine_bboxes = fine_ah.get_local_bboxes();
105 const auto &coarse_bboxes = coarse_ah.get_local_bboxes();
106
107 const IndexSet &locally_owned_dofs_fine =
108 fine_agglo_dh.locally_owned_dofs();
109 const IndexSet locally_relevant_dofs_fine =
110 DoFTools::extract_locally_relevant_dofs(fine_agglo_dh);
111
112 const IndexSet &locally_owned_dofs_coarse =
113 coarse_agglo_dh.locally_owned_dofs();
114
115 std::conditional_t<is_trilinos_matrix,
116 TrilinosWrappers::SparsityPattern,
117 DynamicSparsityPattern>
118 dsp;
119
120 if constexpr (is_trilinos_matrix)
121 dsp.reinit(locally_owned_dofs_fine,
122 locally_owned_dofs_coarse,
123 tria.get_communicator());
124 else
125 dsp.reinit(fine_agglo_dh.n_dofs(),
126 coarse_agglo_dh.n_dofs(),
127 locally_relevant_dofs_fine);
128
129 const unsigned int dofs_per_cell = fe.dofs_per_cell;
130 std::vector<types::global_dof_index> agglo_dof_indices(dofs_per_cell);
131 std::vector<types::global_dof_index> standard_dof_indices(dofs_per_cell);
132 std::vector<types::global_dof_index> output_dof_indices(dofs_per_cell);
133
134 const std::vector<Point<dim>> &unit_support_points =
135 fe.get_unit_support_points();
136 Quadrature<dim> quad(unit_support_points);
137 FEValues<dim, spacedim> output_fe_values(fe,
138 quad,
139 update_quadrature_points);
140
141 std::vector<types::global_dof_index> local_dof_indices_coarse(
142 dofs_per_cell);
143 std::vector<types::global_dof_index> local_dof_indices_child(dofs_per_cell);
144
145 for (const auto &polytope : coarse_ah.polytope_iterators())
146 if (polytope->is_locally_owned())
147 {
148 polytope->get_dof_indices(local_dof_indices_coarse);
149
150 // Get local children and their DoFs
151 const auto &children_polytopes = polytope->children();
152 for (const types::global_cell_index child_idx : children_polytopes)
153 {
154 const typename DoFHandler<dim>::active_cell_iterator &child_dh =
155 fine_ah.polytope_to_dh_iterator(child_idx);
156 child_dh->get_dof_indices(local_dof_indices_child);
157 for (const auto row : local_dof_indices_child)
158 dsp.add_entries(row,
159 local_dof_indices_coarse.begin(),
160 local_dof_indices_coarse.end());
161 }
162 }
163
164 const auto assemble_injection_matrix = [&]() {
165 FullMatrix<NumberType> local_matrix(dofs_per_cell, dofs_per_cell);
166 std::vector<Point<dim>> reference_q_points(dofs_per_cell);
167
168 // Dummy AffineConstraints, only needed for loc2glb
169 AffineConstraints<NumberType> c;
170 c.close();
171
172 for (const auto &polytope : coarse_ah.polytope_iterators())
173 if (polytope->is_locally_owned())
174 {
175 polytope->get_dof_indices(local_dof_indices_coarse);
176 const BoundingBox<dim> &coarse_bbox =
177 coarse_bboxes[polytope->index()];
178
179 // Get local children of the present polytope
180 const auto &children_polytopes = polytope->children();
181 for (const types::global_cell_index child_idx : children_polytopes)
182 {
183 const BoundingBox<dim> &fine_bbox = fine_bboxes[child_idx];
184 const typename DoFHandler<dim>::active_cell_iterator &child_dh =
185 fine_ah.polytope_to_dh_iterator(child_idx);
186 child_dh->get_dof_indices(local_dof_indices_child);
187
188 local_matrix = 0.;
189
190 // compute real location of support points
191 std::vector<Point<dim>> real_qpoints;
192 real_qpoints.reserve(unit_support_points.size());
193 for (const Point<dim> &p : unit_support_points)
194 real_qpoints.push_back(fine_bbox.unit_to_real(p));
195
196 for (unsigned int i = 0; i < local_dof_indices_coarse.size();
197 ++i)
198 {
199 const auto &p = coarse_bbox.real_to_unit(real_qpoints[i]);
200 for (unsigned int j = 0; j < local_dof_indices_child.size();
201 ++j)
202 {
203 local_matrix(i, j) = fe.shape_value(j, p);
204 }
205 }
206
207 c.distribute_local_to_global(local_matrix,
208 local_dof_indices_child,
209 local_dof_indices_coarse,
210 matrix);
211 }
212 }
213 };
214
215
216 if constexpr (is_trilinos_matrix)
217 {
218 dsp.compress();
219 matrix.reinit(dsp);
220 assemble_injection_matrix();
221 }
222 else if constexpr (is_serial_matrix)
223 {
224 sp.copy_from(dsp);
225 matrix.reinit(sp);
226 assemble_injection_matrix();
227 }
228 else
229 {
230 // PETSc types not implemented.
231 (void)coarse_ah;
232 (void)fine_ah;
233 (void)sp;
234 (void)matrix;
235 AssertThrow(false,
236 ExcNotImplemented(
237 "This injection does not support PETSc types."));
238 }
239
240 // If tria is distributed
241 if (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
242 &tria) != nullptr)
243 matrix.compress(VectorOperation::add);
244 }

260 template <typename VectorType, typename MatrixType, typename SolverType>
261 class MGCoarseDirect : public MGCoarseGridBase<VectorType>
262 {
263 public:
264 explicit MGCoarseDirect(const MatrixType &matrix)
265 : direct_solver(TrilinosWrappers::SolverDirect::AdditionalData())
266 , coarse_matrix(matrix)
267 {
268 // Check if matrix types are supported
269 static constexpr bool is_serial_matrix =
270 std::is_same_v<SolverType, SparseDirectUMFPACK>;
271 static constexpr bool is_trilinos_matrix =
272 std::is_same_v<SolverType, TrilinosWrappers::SolverDirect>;
273 [[maybe_unused]] static constexpr bool is_matrix_type_supported =
274 is_serial_matrix || is_trilinos_matrix;
275 Assert(is_matrix_type_supported, ExcNotImplemented());
276
277 // Check on the vector types: standard deal.II vectors, LA::d::V, or
278 // Trilinos vectors.
279 if constexpr (is_serial_matrix)
280 {
281 [[maybe_unused]] static constexpr bool is_serial_vector =
282 std::is_same_v<VectorType,
283 dealii::Vector<typename MatrixType::value_type>>;
284 Assert(is_serial_vector, ExcNotImplemented());
285 }
286 else if constexpr (is_trilinos_matrix)
287 {
288 [[maybe_unused]] static constexpr bool is_supported_parallel_vector =
289 std::is_same_v<VectorType,
290 LinearAlgebra::distributed::Vector<
291 typename MatrixType::value_type>> ||
292 std::is_same_v<VectorType, TrilinosWrappers::MPI::Vector>;
293 Assert(is_supported_parallel_vector, ExcNotImplemented());
294 }
295 else
296 {
297 AssertThrow(false, ExcNotImplemented());
298 // DEAL_II_NOT_IMPLEMENTED(); //not available in older releases
299 }
300
301
302 // regardless of UMFPACK or Trilinos, both direct solvers need a call
303 // to `initialize()`.
304 direct_solver.initialize(coarse_matrix);
305 }

484 const TrilinosWrappers::SparseMatrix &
485 get_system_matrix() const
486 {
487 // Boilerplate for SIP-DG form. TODO: unify interface.
489 const auto cell_operation = [&](auto &phi) {
490 phi.evaluate(EvaluationFlags::gradients);
491 for (unsigned int q = 0; q < phi.n_q_points; ++q)
492 phi.submit_gradient(phi.get_gradient(q), q);
493 phi.integrate(EvaluationFlags::gradients);
494 };
495
496 const auto face_operation = [&](auto &phi_m, auto &phi_p) {
497 phi_m.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
498 phi_p.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
499
500 VectorizedArrayType sigmaF =
501 (std::abs(
502 (phi_m.normal_vector(0) * phi_m.inverse_jacobian(0))[dim - 1]) +
503 std::abs(
504 (phi_m.normal_vector(0) * phi_p.inverse_jacobian(0))[dim - 1])) *
505 (number)(std::max(fe_degree, 1) * (fe_degree + 1.0));
506
507 for (unsigned int q = 0; q < phi_m.n_q_points; ++q)
508 {
509 VectorizedArrayType average_value =
510 (phi_m.get_value(q) - phi_p.get_value(q)) * 0.5;
511 VectorizedArrayType average_valgrad =
512 phi_m.get_normal_derivative(q) + phi_p.get_normal_derivative(q);
513 average_valgrad =
514 average_value * 2. * sigmaF - average_valgrad * 0.5;
515 phi_m.submit_normal_derivative(-average_value, q);
516 phi_p.submit_normal_derivative(-average_value, q);
517 phi_m.submit_value(average_valgrad, q);
518 phi_p.submit_value(-average_valgrad, q);
519 }
520 phi_m.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
521 phi_p.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
522 };
523
524 const auto boundary_operation = [&](auto &phi_m) {
525 phi_m.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
526 VectorizedArrayType sigmaF =
527 std::abs(
528 (phi_m.normal_vector(0) * phi_m.inverse_jacobian(0))[dim - 1]) *
529 number(std::max(fe_degree, 1) * (fe_degree + 1.0)) * 2.0;
530
531 for (unsigned int q = 0; q < phi_m.n_q_points; ++q)
532 {
533 VectorizedArrayType average_value = phi_m.get_value(q);
534 VectorizedArrayType average_valgrad =
535 -phi_m.get_normal_derivative(q);
536 average_valgrad += average_value * sigmaF * 2.0;
537 phi_m.submit_normal_derivative(-average_value, q);
538 phi_m.submit_value(average_valgrad, q);
539 }
540
541 phi_m.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
542 };
543
544
546
547
548 // Check if matrix has already been set up.
549 if (system_matrix.m() == 0 && system_matrix.n() == 0)
550 {
551 // Set up sparsity pattern of system matrix.
552 const auto &dof_handler = data.get_dof_handler();
553
554 TrilinosWrappers::SparsityPattern dsp(
555 data.get_mg_level() != numbers::invalid_unsigned_int ?
556 dof_handler.locally_owned_mg_dofs(data.get_mg_level()) :
557 dof_handler.locally_owned_dofs(),
558 data.get_task_info().communicator);
559
560 if (data.get_mg_level() != numbers::invalid_unsigned_int)
561 MGTools::make_flux_sparsity_pattern(dof_handler,
562 dsp,
563 data.get_mg_level(),
564 constraints);
565 else
566 DoFTools::make_flux_sparsity_pattern(dof_handler, dsp, constraints);
567
568 dsp.compress();
569 system_matrix.reinit(dsp);
570
571 // Assemble system matrix. Notice that degree 1 has been hardcoded.
572
573 MatrixFreeTools::compute_matrix<dim,
574 degree,
575 n_qpoints,
576 n_components,
577 number,
578 VectorizedArrayType>(
579 data,
580 constraints,
581 system_matrix,
582 cell_operation,
583 face_operation,
584 boundary_operation);
585 }
586
587 return system_matrix;
588 }

592 void
593 get_system_matrix(TrilinosWrappers::SparseMatrix &mg_matrix) const
594 {
595 // Boilerplate for SIP-DG form. TODO: unify interface.
597 const auto cell_operation = [&](auto &phi) {
598 phi.evaluate(EvaluationFlags::gradients);
599 for (unsigned int q = 0; q < phi.n_q_points; ++q)
600 phi.submit_gradient(phi.get_gradient(q), q);
601 phi.integrate(EvaluationFlags::gradients);
602 };
603
604 const auto face_operation = [&](auto &phi_m, auto &phi_p) {
605 phi_m.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
606 phi_p.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
607
608 VectorizedArrayType sigmaF =
609 (std::abs(
610 (phi_m.normal_vector(0) * phi_m.inverse_jacobian(0))[dim - 1]) +
611 std::abs(
612 (phi_m.normal_vector(0) * phi_p.inverse_jacobian(0))[dim - 1])) *
613 (number)(std::max(fe_degree, 1) * (fe_degree + 1.0));
614
615 for (unsigned int q = 0; q < phi_m.n_q_points; ++q)
616 {
617 VectorizedArrayType average_value =
618 (phi_m.get_value(q) - phi_p.get_value(q)) * 0.5;
619 VectorizedArrayType average_valgrad =
620 phi_m.get_normal_derivative(q) + phi_p.get_normal_derivative(q);
621 average_valgrad =
622 average_value * 2. * sigmaF - average_valgrad * 0.5;
623 phi_m.submit_normal_derivative(-average_value, q);
624 phi_p.submit_normal_derivative(-average_value, q);
625 phi_m.submit_value(average_valgrad, q);
626 phi_p.submit_value(-average_valgrad, q);
627 }
628 phi_m.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
629 phi_p.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
630 };
631
632 const auto boundary_operation = [&](auto &phi_m) {
633 phi_m.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
634 VectorizedArrayType sigmaF =
635 std::abs(
636 (phi_m.normal_vector(0) * phi_m.inverse_jacobian(0))[dim - 1]) *
637 number(std::max(fe_degree, 1) * (fe_degree + 1.0)) * 2.0;
638
639 for (unsigned int q = 0; q < phi_m.n_q_points; ++q)
640 {
641 VectorizedArrayType average_value = phi_m.get_value(q);
642 VectorizedArrayType average_valgrad =
643 -phi_m.get_normal_derivative(q);
644 average_valgrad += average_value * sigmaF * 2.0;
645 phi_m.submit_normal_derivative(-average_value, q);
646 phi_m.submit_value(average_valgrad, q);
647 }
648
649 phi_m.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
650 };
651
652
654
655 // Check if matrix has already been set up.
656 AssertThrow((mg_matrix.m() == 0 && mg_matrix.n() == 0),
657 ExcInternalError());
658
659 // Set up sparsity pattern of system matrix.
660 const DoFHandler<dim> &dof_handler = data.get_dof_handler();
661
662 const IndexSet &system_partitioning = dof_handler.locally_owned_dofs();
663 const IndexSet system_relevant_set =
664 DoFTools::extract_locally_relevant_dofs(dof_handler);
665
666
667 DynamicSparsityPattern dsp(dof_handler.n_dofs(),
668 dof_handler.n_dofs(),
669 system_relevant_set);
670 DoFTools::make_flux_sparsity_pattern(dof_handler, dsp);
671
672 SparsityTools::distribute_sparsity_pattern(
673 dsp,
674 system_partitioning,
675 data.get_task_info().communicator,
676 system_relevant_set);
677 mg_matrix.reinit(system_partitioning,
678 dsp,
679 data.get_task_info().communicator);
680
681 // Assemble system matrix.
682 MatrixFreeTools::compute_matrix<dim,
683 degree,
684 n_qpoints,
685 n_components,
686 number,
687 VectorizedArrayType>(data,
688 constraints,
689 mg_matrix,
690 cell_operation,
691 face_operation,
692 boundary_operation);
693 }

697 void
698 rhs(LinearAlgebra::distributed::Vector<number> &b) const
699 {
700 const int dummy = 0;
701
702
703 data.template cell_loop<LinearAlgebra::distributed::Vector<number>, int>(
704 [](const auto &matrix_free, auto &dst, const auto &, const auto cells) {
705 FEEvaluation<dim, -1, 0, n_components, number> phi(matrix_free,
706 cells);
707 for (unsigned int cell = cells.first; cell < cells.second; ++cell)
708 {
709 phi.reinit(cell);
710 for (unsigned int q = 0; q < phi.n_q_points; ++q)
711 {
712 if constexpr (n_components == 1)
713 {
714 phi.submit_value(1.0, q);
715 }
716 else
717 {
718 Tensor<1, n_components, VectorizedArray<number>> temp;
719 for (unsigned int v = 0;
720 v < VectorizedArray<number>::size();
721 ++v)
722 {
723 for (unsigned int i = 0; i < n_components; i++)
724 temp[i][v] = 1.;
725 }
726 phi.submit_value(temp, q);
727 }
728 }
729
730 phi.integrate_scatter(EvaluationFlags::values, dst);
731 }
732 },
733 b,

785 const MatrixFree<dim, number> &data,
786 LinearAlgebra::distributed::Vector<number> &dst,
787 const LinearAlgebra::distributed::Vector<number> &src,
788 const std::pair<unsigned int, unsigned int> &face_range) const
789 {
790 FEFaceEvaluation<dim, -1, 0, 1, number> fe_eval(data, true);
791 FEFaceEvaluation<dim, -1, 0, 1, number> fe_eval_neighbor(data, false);
792
793 for (unsigned int face = face_range.first; face < face_range.second;
794 ++face)
795 {
796 fe_eval.reinit(face);
797 fe_eval_neighbor.reinit(face);
798
799 fe_eval.read_dof_values(src);
800 fe_eval.evaluate(EvaluationFlags::values |
801 EvaluationFlags::gradients);
802 fe_eval_neighbor.read_dof_values(src);
803 fe_eval_neighbor.evaluate(EvaluationFlags::values |
804 EvaluationFlags::gradients);
805 VectorizedArray<number> sigmaF =
806 (std::abs((fe_eval.normal_vector(0) *
807 fe_eval.inverse_jacobian(0))[dim - 1]) +
808 std::abs((fe_eval.normal_vector(0) *
809 fe_eval_neighbor.inverse_jacobian(0))[dim - 1])) *
810 (number)(std::max(fe_degree, 1) * (fe_degree + 1.0));
811
812 for (unsigned int q = 0; q < fe_eval.n_q_points; ++q)
813 {
814 VectorizedArray<number> average_value =
815 (fe_eval.get_value(q) - fe_eval_neighbor.get_value(q)) * 0.5;
816 VectorizedArray<number> average_valgrad =
817 fe_eval.get_normal_derivative(q) +
818 fe_eval_neighbor.get_normal_derivative(q);
819 average_valgrad =
820 average_value * 2. * sigmaF - average_valgrad * 0.5;
821 fe_eval.submit_normal_derivative(-average_value, q);
822 fe_eval_neighbor.submit_normal_derivative(-average_value, q);
823 fe_eval.submit_value(average_valgrad, q);
824 fe_eval_neighbor.submit_value(-average_valgrad, q);
825 }
826 fe_eval.integrate(EvaluationFlags::values |
827 EvaluationFlags::gradients);
828 fe_eval.distribute_local_to_global(dst);
829 fe_eval_neighbor.integrate(EvaluationFlags::values |
830 EvaluationFlags::gradients);
831 fe_eval_neighbor.distribute_local_to_global(dst);
832 }
