57 template <int dim, int spacedim, typename MatrixType>
58 void
59 fill_injection_matrix(const AgglomerationHandler<dim, spacedim> &coarse_ah,
60 const AgglomerationHandler<dim, spacedim> &fine_ah,
61 SparsityPattern &sp,
62 MatrixType &matrix)
63 {
64 using NumberType = typename MatrixType::value_type;
65 // First, check that we support the matrix types
66 static constexpr bool is_trilinos_matrix =
67 std::is_same_v<TrilinosWrappers::SparseMatrix, MatrixType>;
68 static constexpr bool is_serial_matrix =
69 std::is_same_v<SparseMatrix<NumberType>, MatrixType>;
70 static constexpr bool is_supported_matrix =
71 is_trilinos_matrix || is_serial_matrix;
72 static_assert(is_supported_matrix);
73 if constexpr (is_trilinos_matrix)
74 Assert(matrix.m() == 0, ExcInternalError());
75 else if constexpr (is_serial_matrix)
76 Assert(sp.empty() && matrix.empty(),
77 ExcMessage(
78 "The destination matrix and its sparsity pattern must the empty "
79 "upon calling this function."));
80
81 Assert(coarse_ah.n_dofs() < fine_ah.n_dofs(), ExcInternalError());
82 AssertDimension(dim, spacedim);
83
84 // Get information from the handlers
85 const DoFHandler<dim, spacedim> &coarse_agglo_dh = coarse_ah.agglo_dh;
86 const DoFHandler<dim, spacedim> &fine_agglo_dh = fine_ah.agglo_dh;
87
88 const FiniteElement<dim, spacedim> &fe = coarse_ah.get_fe();
89 const Triangulation<dim, spacedim> &tria = coarse_ah.get_triangulation();
90 const auto &fine_bboxes = fine_ah.get_local_bboxes();
91 const auto &coarse_bboxes = coarse_ah.get_local_bboxes();
92
93 const IndexSet &locally_owned_dofs_fine =
94 fine_agglo_dh.locally_owned_dofs();
95 const IndexSet locally_relevant_dofs_fine =
96 DoFTools::extract_locally_relevant_dofs(fine_agglo_dh);
97
98 const IndexSet &locally_owned_dofs_coarse =
99 coarse_agglo_dh.locally_owned_dofs();
100
101 std::conditional_t<is_trilinos_matrix,
102 TrilinosWrappers::SparsityPattern,
103 DynamicSparsityPattern>
104 dsp;
105
106 if constexpr (is_trilinos_matrix)
107 dsp.reinit(locally_owned_dofs_fine,
108 locally_owned_dofs_coarse,
109 tria.get_communicator());
110 else
111 dsp.reinit(fine_agglo_dh.n_dofs(),
112 coarse_agglo_dh.n_dofs(),
113 locally_relevant_dofs_fine);
114
115 const unsigned int dofs_per_cell = fe.dofs_per_cell;
116 std::vector<types::global_dof_index> agglo_dof_indices(dofs_per_cell);
117 std::vector<types::global_dof_index> standard_dof_indices(dofs_per_cell);
118 std::vector<types::global_dof_index> output_dof_indices(dofs_per_cell);
119
120 const std::vector<Point<dim>> &unit_support_points =
121 fe.get_unit_support_points();
122 Quadrature<dim> quad(unit_support_points);
123 FEValues<dim, spacedim> output_fe_values(fe,
124 quad,
125 update_quadrature_points);
126
127 std::vector<types::global_dof_index> local_dof_indices_coarse(
128 dofs_per_cell);
129 std::vector<types::global_dof_index> local_dof_indices_child(dofs_per_cell);
130
131 for (const auto &polytope : coarse_ah.polytope_iterators())
132 if (polytope->is_locally_owned())
133 {
134 polytope->get_dof_indices(local_dof_indices_coarse);
135
136 // Get local children and their DoFs
137 const auto &children_polytopes = polytope->children();
138 for (const types::global_cell_index child_idx : children_polytopes)
139 {
140 const typename DoFHandler<dim>::active_cell_iterator &child_dh =
141 fine_ah.polytope_to_dh_iterator(child_idx);
142 child_dh->get_dof_indices(local_dof_indices_child);
143 for (const auto row : local_dof_indices_child)
144 dsp.add_entries(row,
145 local_dof_indices_coarse.begin(),
146 local_dof_indices_coarse.end());
147 }
148 }
149
150 const auto assemble_injection_matrix = [&]() {
151 FullMatrix<NumberType> local_matrix(dofs_per_cell, dofs_per_cell);
152 std::vector<Point<dim>> reference_q_points(dofs_per_cell);
153
154 // Dummy AffineConstraints, only needed for loc2glb
155 AffineConstraints<NumberType> c;
156 c.close();
157
158 for (const auto &polytope : coarse_ah.polytope_iterators())
159 if (polytope->is_locally_owned())
160 {
161 polytope->get_dof_indices(local_dof_indices_coarse);
162 const BoundingBox<dim> &coarse_bbox =
163 coarse_bboxes[polytope->index()];
164
165 // Get local children of the present polytope
166 const auto &children_polytopes = polytope->children();
167 for (const types::global_cell_index child_idx : children_polytopes)
168 {
169 const BoundingBox<dim> &fine_bbox = fine_bboxes[child_idx];
170 const typename DoFHandler<dim>::active_cell_iterator &child_dh =
171 fine_ah.polytope_to_dh_iterator(child_idx);
172 child_dh->get_dof_indices(local_dof_indices_child);
173
174 local_matrix = 0.;
175
176 // compute real location of support points
177 std::vector<Point<dim>> real_qpoints;
178 real_qpoints.reserve(unit_support_points.size());
179 for (const Point<dim> &p : unit_support_points)
180 real_qpoints.push_back(fine_bbox.unit_to_real(p));
181
182 for (unsigned int i = 0; i < local_dof_indices_coarse.size();
183 ++i)
184 {
185 const auto &p = coarse_bbox.real_to_unit(real_qpoints[i]);
186 for (unsigned int j = 0; j < local_dof_indices_child.size();
187 ++j)
188 {
189 local_matrix(i, j) = fe.shape_value(j, p);
190 }
191 }
192
193 c.distribute_local_to_global(local_matrix,
194 local_dof_indices_child,
195 local_dof_indices_coarse,
196 matrix);
197 }
198 }
199 };
200
201
202 if constexpr (is_trilinos_matrix)
203 {
204 dsp.compress();
205 matrix.reinit(dsp);
206 assemble_injection_matrix();
207 }
208 else if constexpr (is_serial_matrix)
209 {
210 sp.copy_from(dsp);
211 matrix.reinit(sp);
212 assemble_injection_matrix();
213 }
214 else
215 {
216 // PETSc types not implemented.
217 (void)coarse_ah;
218 (void)fine_ah;
219 (void)sp;
220 (void)matrix;
221 AssertThrow(false,
222 ExcNotImplemented(
223 "This injection does not support PETSc types."));
224 }
225
226 // If tria is distributed
227 if (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
228 &tria) != nullptr)
229 matrix.compress(VectorOperation::add);
230 }
231
232} // namespace Utils
