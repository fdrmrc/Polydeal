240 template <class FiniteElement>
241 void
242 distribute_agglomerated_dofs(const FiniteElement &fe_space)
243 {
244 Assert((dynamic_cast<const FE_DGQ<dim, spacedim> *>(&fe_space)),
245 ExcNotImplemented(
246 "Currently, this interface supports only DG discretizations."));
247 fe = std::make_unique<FE_DGQ<dim, spacedim>>(fe_space);
248
249 if (hybrid_mesh)
250 {
251 // the mesh is composed by standard and agglomerate cells. initialize
252 // classes needed for standard cells in order to treat that finite
253 // element space as defined on a standard shape and not on the
254 // BoundingBox.
255 standard_scratch =
256 std::make_unique<ScratchData>(*mapping,
257 *fe,
258 QGauss<dim>(2 * fe_space.degree + 2),
259 internal_agglomeration_flags);
260 }
261
262
263 fe_collection.push_back(*fe); // master
264 fe_collection.push_back(FE_Nothing<dim, spacedim>()); // slave
265 fe_collection.push_back(*fe); // standard
266
267 initialize_hp_structure();
268
269 // in case the tria is distributed, communicate ghost information with
270 // neighboring ranks
271 const bool needs_ghost_info =
272 dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
273 &*tria) != nullptr;
274 if (needs_ghost_info)
275 setup_ghost_polytopes();
276
277 setup_connectivity_of_agglomeration();
278
279 if (needs_ghost_info)
280 exchange_interface_values();
281 }
