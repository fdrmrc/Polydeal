338 unsigned int
339 n_agglomerated_faces_per_cell(
340 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell)
341 const
342 {
343 unsigned int n_neighbors = 0;
344 for (const auto &f : cell->face_indices())
345 {
346 const auto &neighboring_cell = cell->neighbor(f);
347 if ((cell->face(f)->at_boundary()) ||
348 (neighboring_cell->is_active() &&
349 !are_cells_agglomerated(cell, neighboring_cell)))
350 {
351 ++n_neighbors;
352 }
353 }
354 return n_neighbors;
355 }

365 unsigned int
366 n_faces(
367 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell) const
368 {
369 Assert(!is_slave_cell(cell), ExcMessage("You cannot pass a slave cell."));
370 if (is_standard_cell(cell))
371 {
372 return cell->n_faces();
373 }
374 else
375 {
376 const auto & agglomeration = get_agglomerate(cell);
377 unsigned int n_neighbors = 0;
378 for (const auto &cell : agglomeration)
379 {
380 n_neighbors += n_agglomerated_faces_per_cell(cell);
381 }
382 return n_neighbors;
383 }

392 typename DoFHandler<dim, spacedim>::active_cell_iterator
393 agglomerated_neighbor(
394 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
395 const unsigned int f) const
396 {
397 Assert(!is_slave_cell(cell),
398 ExcMessage("This cells is not supposed to be a slave."));
399 if (!at_boundary(cell, f))
400 {
401 if (is_standard_cell(cell) && is_standard_cell(cell->neighbor(f)))
402 {
403 return cell->neighbor(f);
404 }
405 else if (is_master_cell(cell))
406 {
407 const auto &neigh = std::get<1>(master_neighbors[{cell, f}]);
408 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
409 *neigh, &agglo_dh);
410 return cell_dh;
411 }
412 else
413 {
414 // If I fall here, I want to find the neighbor for a standard cell
415 // adjacent to an agglomeration
416 const auto &master_cell =
417 master_slave_relationships_iterators[cell->neighbor(f)
418 ->active_cell_index()];
419 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
420 *master_cell, &agglo_dh);
421 return cell_dh;
422 }
423 }
424 else
425 {
426 return {};
427 }
428 }

437 unsigned int
438 neighbor_of_agglomerated_neighbor(
439 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
440 const unsigned int f) const
441 {
442 Assert(!is_slave_cell(cell),
443 ExcMessage("This cells is not supposed to be a slave."));
444 // First, make sure it's not a boundary face.
445 if (!at_boundary(cell, f))
446 {
447 if (is_standard_cell(cell) && is_standard_cell(cell->neighbor(f)))
448 {
449 return cell->neighbor_of_neighbor(f);
450 }
451 else if (is_master_cell(cell) &&
452 is_master_cell(agglomerated_neighbor(cell, f)))
453 {
454 const auto &current_agglo_info = master_neighbors[{cell, f}];
455 const auto &local_f_idx = std::get<0>(current_agglo_info);
456 const auto &current_cell = std::get<3>(current_agglo_info);
457
458 const auto &agglo_neigh =
459 agglomerated_neighbor(cell,
460 f); // returns the neighboring master
461 const unsigned int n_faces_agglomerated_neighbor =
462 n_faces(agglo_neigh);
463 // Loop over all cells of neighboring agglomerate
464 for (unsigned int f_out = 0; f_out < n_faces_agglomerated_neighbor;
465 ++f_out)
466 {
467 if (agglomerated_neighbor(agglo_neigh, f_out).state() ==
468 IteratorState::valid &&
469 current_cell->neighbor(local_f_idx).state() ==
470 IteratorState::valid &&
471 current_cell.state() == IteratorState::valid)
472 {
473 const auto &neighboring_agglo_info =
474 master_neighbors[{agglo_neigh, f_out}];
475 const auto &local_f_out_idx =
476 std::get<0>(neighboring_agglo_info);
477 const auto &current_cell_out =
478 std::get<3>(neighboring_agglo_info);
479 const auto &other_standard =
480 std::get<1>(neighboring_agglo_info);
481
482 // Here, an extra condition is needed because there can be
483 // more than one face index that returns the same neighbor
484 // if you simply check who is f' s.t.
485 // cell->neigh(f)->neigh(f') == cell. Hence, an extra
486 // condition must be added.
487
488 if (other_standard.state() == IteratorState::valid &&
489 agglomerated_neighbor(agglo_neigh, f_out)
490 ->active_cell_index() ==
491 cell->active_cell_index() &&
492 current_cell->active_cell_index() ==
493 current_cell_out->neighbor(local_f_out_idx)
494 ->active_cell_index() &&
495 current_cell->neighbor(local_f_idx) == current_cell_out)
496 return f_out;
497 }
498 }
499 Assert(false, ExcInternalError());
500 return {}; // just to suppress warnings
501 }
502 else if (is_master_cell(cell) &&
503 is_standard_cell(agglomerated_neighbor(cell, f)))
504 {
505 return std::get<2>(master_neighbors[{cell, f}]);
506 }
507 else
508 {
509 // If I fall here, I want to find the neighbor of neighbor for a
510 // standard cell adjacent to an agglomeration.
511 const auto &master_cell = agglomerated_neighbor(
512 cell, f); // this is the master of the neighboring agglomeration
513
514 return shared_face_agglomeration_idx[{master_cell, cell, f}];
515 }
516 }
517 else
518 {
519 // Face is at boundary
520 return numbers::invalid_unsigned_int;
521 }
522 }

973 void
974 create_bounding_box(const AgglomerationContainer & polytope,
975 const types::global_cell_index master_idx)
976 {
977 Assert(n_agglomerations > 0,
978 ExcMessage("No agglomeration has been performed."));
979 Assert(dim > 1, ExcNotImplemented());
980
981 std::vector<types::global_dof_index> dof_indices(euler_fe->dofs_per_cell);
982 std::vector<Point<spacedim>> pts; // store all the vertices
983 for (const auto &cell : polytope)
984 {
985 typename DoFHandler<dim, spacedim>::cell_iterator cell_dh(*cell,
986 &euler_dh);
987 cell_dh->get_dof_indices(dof_indices);

1119 setup_master_neighbor_connectivity(
1120 const typename Triangulation<dim, spacedim>::active_cell_iterator
1121 &master_cell) const
1122 {
1123 Assert(master_slave_relationships[master_cell->active_cell_index()] == -1,
1124 ExcMessage("The present cell is not a master one."));
1125
1126 const auto &agglomeration = get_agglomerate(master_cell);
1127
1128 unsigned int n_agglo_faces = 0;
1129 for (const auto &cell : agglomeration)
1130 {
1131 for (const auto f : cell->face_indices())
1132 {
1133 const auto &neighboring_cell = cell->neighbor(f);
1134 // Check if cell is not on the boundary and if it's not
1135 // agglomerated with the neighbor If so, it's a neighbor of the
1136 // present agglomeration
1137 if (neighboring_cell.state() == IteratorState::valid &&
1138 !are_cells_agglomerated(cell, neighboring_cell))
1139 {
1140 // a new face of the agglomeration has been discovered.
1141 polygon_boundary[master_cell].push_back(cell->face(f));
1142
1143 const auto &cell_and_face =
1144 CellAndFace(master_cell, n_agglo_faces); //(agglo,f)
1145 const auto nof = cell->neighbor_of_neighbor(f); // loc(f')
1146
1147 if (is_slave_cell(neighboring_cell))
1148 master_neighbors.emplace(
1149 cell_and_face,
1150 std::make_tuple(f,
1151 master_slave_relationships_iterators
1152 [neighboring_cell->active_cell_index()],
1153 nof,
1154 cell));
1155 else
1156 master_neighbors.emplace(
1157 cell_and_face,
1158 std::make_tuple(
1159 f,
1160 neighboring_cell,
1161 nof,
1162 cell)); //(agglo,f)
1163 //->(loc(f),other_deal_cell,loc(f'),dealcell)
1164
1165 // Now, link the index of the agglomerated face with the
1166 // master and the neighboring cell.
1167 shared_face_agglomeration_idx.emplace(
1168 MasterAndNeighborAndFace(master_cell, neighboring_cell, nof),
1169 n_agglo_faces);
1170 ++n_agglo_faces;
1171 }
1172 else if (cell->face(f)->at_boundary())
1173 {
1174 // Boundary face of a boundary cell.
1175 // Note that the neighboring cell must be invalid.
1176 const auto &cell_and_face =
1177 CellAndFace(master_cell, n_agglo_faces);
1178
1179 polygon_boundary[master_cell].push_back(cell->face(f));
1180
1181 master_neighbors.emplace(
1182 cell_and_face,
1183 std::make_tuple(f,
1184 neighboring_cell,
1185 std::numeric_limits<unsigned int>::max(),
1186 cell)); // TODO: check what the last element
1187 // should be...
1188 ++n_agglo_faces;
1189 }
1190 }
1191 }

1292 public:

1294 reinit_master(
1295 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
1296 const unsigned int face_index,
1297 std::unique_ptr<NonMatching::FEImmersedSurfaceValues<spacedim>>
1298 & agglo_isv_ptr,
1299 const AgglomerationHandler<dim, spacedim> &handler)
1300 {
1301 Assert(handler.is_master_cell(cell),
1302 ExcMessage("This cell must be a master one."));
1303 const auto &info_neighbors =
1304 handler.master_neighbors[{cell, face_index}];
1305 const auto &local_face_idx = std::get<0>(info_neighbors);
1306 const auto &neighboring_cell = std::get<3>(info_neighbors);
1307
1308 if (neighboring_cell.state() == IteratorState::valid)
1309 {
1310 handler.no_face_values->reinit(neighboring_cell, local_face_idx);
1311 auto q_points = handler.no_face_values->get_quadrature_points();
1312
1313 const auto &JxWs = handler.no_face_values->get_JxW_values();
1314 const auto &normals = handler.no_face_values->get_normal_vectors();
1315
1316 const auto &bbox =
1317 handler
1318 .bboxes[handler.master2polygon.at(cell->active_cell_index())];
1319 const double bbox_measure = bbox.volume();
1320 std::vector<Point<spacedim>> unit_q_points;
1321
1322 std::transform(q_points.begin(),
1323 q_points.end(),
1324 std::back_inserter(unit_q_points),
1325 [&](const Point<spacedim> &p) {
1326 return bbox.real_to_unit(p);
1327 });
1328
1329 // Weights must be scaled with det(J)*|J^-t n| for each quadrature
1330 // point. Use the fact that we are using a BBox, so the jacobi
1331 // entries are the side_length in each direction. Unfortunately,
1332 // normal vectors will be scaled internally by deal.II by using a
1333 // covariant transformation. In order not to change the normals, we
1334 // multiply by the correct factors in order to obtain the original
1335 // normal after the call to `reinit(cell)`.
1336 std::vector<double> scale_factors(q_points.size());
1337 std::vector<double> scaled_weights(q_points.size());
1338 std::vector<Tensor<1, dim>> scaled_normals(q_points.size());
1339
1340 for (unsigned int q = 0; q < q_points.size(); ++q)
1341 {
1342 for (unsigned int direction = 0; direction < spacedim;
1343 ++direction)
1344 scaled_normals[q][direction] =
1345 normals[q][direction] * (bbox.side_length(direction));
1346
1347 scaled_weights[q] =
1348 (JxWs[q] * scaled_normals[q].norm()) / bbox_measure;
1349 scaled_normals[q] /= scaled_normals[q].norm();
1350 }
1351
1352 NonMatching::ImmersedSurfaceQuadrature<dim, spacedim> surface_quad(
1353 unit_q_points, scaled_weights, scaled_normals);
1354
1355 agglo_isv_ptr =
1356 std::make_unique<NonMatching::FEImmersedSurfaceValues<spacedim>>(
1357 *(handler.euler_mapping),
1358 *(handler.fe),
1359 surface_quad,
1360 handler.agglomeration_face_flags);
1361
1362 agglo_isv_ptr->reinit(cell);
1363
1364 return *agglo_isv_ptr;
1365 }
1366 else
1367 {
1368 // Then it's a boundary face of an agglomeration living on the
1369 // boundary of the tria. You need to return an FEFaceValues on the
1370 // boundary face of a boundary cell.
1371 handler.no_face_values->reinit(neighboring_cell, local_face_idx);
1372
1373 // TODO: check if *mapping or *euler_mapping

1377 handler.fe_collection[2],
1378 handler.agglomeration_quad,
1379 handler.agglomeration_flags,
1380 handler.agglomeration_face_quad,
1381 handler.agglomeration_face_flags);
1382

1384 }
