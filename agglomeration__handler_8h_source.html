296 unsigned int
297 n_agglomerated_faces_per_cell(
298 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell)
299 const
300 {
301 unsigned int n_neighbors = 0;
302 for (const auto &f : cell->face_indices())
303 {
304 const auto &neighboring_cell = cell->neighbor(f);
305 if ((cell->face(f)->at_boundary()) ||
306 (neighboring_cell->is_active() &&
307 !are_cells_agglomerated(cell, neighboring_cell)))
308 {
309 ++n_neighbors;
310 }
311 }
312 return n_neighbors;
313 }

323 unsigned int
324 n_faces(
325 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell) const
326 {
327 Assert(!is_slave_cell(cell), ExcMessage("You cannot pass a slave cell."));
328 if (is_standard_cell(cell))
329 {
330 return cell->n_faces();
331 }
332 else
333 {
334 const auto & agglomeration = get_agglomerate(cell);
335 unsigned int n_neighbors = 0;
336 for (const auto &cell : agglomeration)
337 {
338 n_neighbors += n_agglomerated_faces_per_cell(cell);
339 }
340 return n_neighbors;
341 }

350 typename DoFHandler<dim, spacedim>::active_cell_iterator
351 agglomerated_neighbor(
352 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
353 const unsigned int f) const
354 {
355 Assert(!is_slave_cell(cell),
356 ExcMessage("This cells is not supposed to be a slave."));
357 if (!at_boundary(cell, f))
358 {
359 if (is_standard_cell(cell) && is_standard_cell(cell->neighbor(f)))
360 {
361 return cell->neighbor(f);
362 }
363 else if (is_master_cell(cell))
364 {
365 const auto &neigh = std::get<1>(master_neighbors[{cell, f}]);
366 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
367 *neigh, &agglo_dh);
368 return cell_dh;
369 }
370 else
371 {
372 // If I fall here, I want to find the neighbor for a standard cell
373 // adjacent to an agglomeration
374 const auto &master_cell =
375 master_slave_relationships_iterators[cell->neighbor(f)
376 ->active_cell_index()];
377 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
378 *master_cell, &agglo_dh);
379 return cell_dh;
380 }
381 }
382 else
383 {
384 return {};
385 }
386 }

395 unsigned int
396 neighbor_of_agglomerated_neighbor(
397 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
398 const unsigned int f) const
399 {
400 Assert(!is_slave_cell(cell),
401 ExcMessage("This cells is not supposed to be a slave."));
402 // First, make sure it's not a boundary face.
403 if (!at_boundary(cell, f))
404 {
405 if (is_standard_cell(cell) && is_standard_cell(cell->neighbor(f)))
406 {
407 return cell->neighbor_of_neighbor(f);
408 }
409 else if (is_master_cell(cell) &&
410 is_master_cell(agglomerated_neighbor(cell, f)))
411 {
412 const auto &current_agglo_info = master_neighbors[{cell, f}];
413 const auto &local_f_idx = std::get<0>(current_agglo_info);
414 const auto &current_cell = std::get<3>(current_agglo_info);
415
416 const auto &agglo_neigh =
417 agglomerated_neighbor(cell,
418 f); // returns the neighboring master
419 const unsigned int n_faces_agglomerated_neighbor =
420 n_faces(agglo_neigh);
421 // Loop over all cells of neighboring agglomerate
422 for (unsigned int f_out = 0; f_out < n_faces_agglomerated_neighbor;
423 ++f_out)
424 {
425 if (agglomerated_neighbor(agglo_neigh, f_out).state() ==
426 IteratorState::valid &&
427 current_cell->neighbor(local_f_idx).state() ==
428 IteratorState::valid &&
429 current_cell.state() == IteratorState::valid)
430 {
431 const auto &neighboring_agglo_info =
432 master_neighbors[{agglo_neigh, f_out}];
433 const auto &local_f_out_idx =
434 std::get<0>(neighboring_agglo_info);
435 const auto &current_cell_out =
436 std::get<3>(neighboring_agglo_info);
437 const auto &other_standard =
438 std::get<1>(neighboring_agglo_info);
439
440 // Here, an extra condition is needed because there can be
441 // more than one face index that returns the same neighbor
442 // if you simply check who is f' s.t.
443 // cell->neigh(f)->neigh(f') == cell. Hence, an extra
444 // condition must be added.
445
446 if (other_standard.state() == IteratorState::valid &&
447 agglomerated_neighbor(agglo_neigh, f_out)
448 ->active_cell_index() ==
449 cell->active_cell_index() &&
450 current_cell->active_cell_index() ==
451 current_cell_out->neighbor(local_f_out_idx)
452 ->active_cell_index() &&
453 current_cell->neighbor(local_f_idx) == current_cell_out)
454 return f_out;
455 }
456 }
457 Assert(false, ExcInternalError());
458 return {}; // just to suppress warnings
459 }
460 else if (is_master_cell(cell) &&
461 is_standard_cell(agglomerated_neighbor(cell, f)))
462 {
463 return std::get<2>(master_neighbors[{cell, f}]);
464 }
465 else
466 {
467 // If I fall here, I want to find the neighbor of neighbor for a
468 // standard cell adjacent to an agglomeration.
469 const auto &master_cell = agglomerated_neighbor(
470 cell, f); // this is the master of the neighboring agglomeration
471
472 return shared_face_agglomeration_idx[{master_cell, cell, f}];
473 }
474 }
475 else
476 {
477 // Face is at boundary
478 return numbers::invalid_unsigned_int;
479 }
480 }

911 void
912 create_bounding_box(
913 const std::vector<
914 typename Triangulation<dim, spacedim>::active_cell_iterator>
915 & vec_of_cells,
916 const types::global_cell_index master_idx)
917 {
918 Assert(n_agglomerations > 0,
919 ExcMessage("No agglomeration has been performed."));
920 Assert(dim > 1, ExcNotImplemented());
921
922 std::vector<types::global_dof_index> dof_indices(euler_fe->dofs_per_cell);
923 std::vector<Point<spacedim>> pts; // store all the vertices
924 for (const auto &cell : vec_of_cells)
925 {
926 typename DoFHandler<dim, spacedim>::cell_iterator cell_dh(*cell,
927 &euler_dh);
928 cell_dh->get_dof_indices(dof_indices);
929 for (const auto i : cell->vertex_indices())
930 pts.push_back(cell->vertex(i));
931 }
932
933 bboxes.emplace_back(pts);
934
935 const auto &p0 =
936 bboxes[master2polygon.at(master_idx)].get_boundary_points().first;
937 const auto &p1 =
938 bboxes[master2polygon.at(master_idx)].get_boundary_points().second;
939 if constexpr (dim == 2)
940 {
941 euler_vector[dof_indices[0]] = p0[0];
942 euler_vector[dof_indices[4]] = p0[1];

1049 void
1050 setup_master_neighbor_connectivity(
1051 const typename Triangulation<dim, spacedim>::active_cell_iterator
1052 &master_cell) const
1053 {
1054 Assert(master_slave_relationships[master_cell->active_cell_index()] == -1,
1055 ExcMessage("The present cell is not a master one."));
1056
1057 const auto &agglomeration = get_agglomerate(master_cell);
1058
1059 unsigned int n_agglo_faces = 0;
1060 for (const auto &cell : agglomeration)
1061 {
1062 for (const auto f : cell->face_indices())
1063 {
1064 const auto &neighboring_cell = cell->neighbor(f);
1065 // Check if cell is not on the boundary and if it's not
1066 // agglomerated with the neighbor If so, it's a neighbor of the
1067 // present agglomeration
1068 if (neighboring_cell.state() == IteratorState::valid &&
1069 !are_cells_agglomerated(cell, neighboring_cell))
1070 {
1071 // a new face of the agglomeration has been discovered.
1072 polygon_boundary[master_cell].push_back(cell->face(f));
1073
1074 const auto &cell_and_face =
1075 CellAndFace(master_cell, n_agglo_faces); //(agglo,f)
1076 const auto nof = cell->neighbor_of_neighbor(f); // loc(f')
1077
1078 if (is_slave_cell(neighboring_cell))
1079 master_neighbors.emplace(
1080 cell_and_face,
1081 std::make_tuple(f,
1082 master_slave_relationships_iterators
1083 [neighboring_cell->active_cell_index()],
1084 nof,
1085 cell));
1086 else
1087 master_neighbors.emplace(
1088 cell_and_face,
1089 std::make_tuple(
1090 f,
1091 neighboring_cell,
1092 nof,
1093 cell)); //(agglo,f)
1094 //->(loc(f),other_deal_cell,loc(f'),dealcell)
1095
1096 // Now, link the index of the agglomerated face with the
1097 // master and the neighboring cell.
1098 shared_face_agglomeration_idx.emplace(
1099 MasterAndNeighborAndFace(master_cell, neighboring_cell, nof),
1100 n_agglo_faces);
1101 ++n_agglo_faces;
1102 }
1103 else if (cell->face(f)->at_boundary())
1104 {
1105 // Boundary face of a boundary cell.
1106 // Note that the neighboring cell must be invalid.
1107 const auto &cell_and_face =
1108 CellAndFace(master_cell, n_agglo_faces);
1109
1110 polygon_boundary[master_cell].push_back(cell->face(f));
1111
1112 master_neighbors.emplace(
1113 cell_and_face,
1114 std::make_tuple(f,
1115 neighboring_cell,
1116 std::numeric_limits<unsigned int>::max(),
1117 cell)); // TODO: check what the last element
1118 // should be...
1119 ++n_agglo_faces;
1120 }
1121 }
1122 }
1123 }
