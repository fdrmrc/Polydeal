226 template <class FiniteElement>
227 void
228 distribute_agglomerated_dofs(const FiniteElement &fe_space)
229 {
230 Assert((dynamic_cast<const FE_DGQ<dim, spacedim> *>(&fe_space)),
231 ExcNotImplemented(
232 "Currently, this interface supports only DG discretizations."));
233 fe = std::make_unique<FE_DGQ<dim, spacedim>>(fe_space);
234
235 fe_collection.push_back(*fe); // master
236 fe_collection.push_back(FE_Nothing<dim, spacedim>()); // slave
237 fe_collection.push_back(*fe); // standard
238

241 }
