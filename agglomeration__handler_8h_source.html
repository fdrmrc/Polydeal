347 unsigned int
348 n_agglomerated_faces_per_cell(
349 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell)
350 const
351 {
352 unsigned int n_neighbors = 0;
353 for (const auto &f : cell->face_indices())
354 {
355 const auto &neighboring_cell = cell->neighbor(f);
356 if ((cell->face(f)->at_boundary()) ||
357 (neighboring_cell->is_active() &&
358 !are_cells_agglomerated(cell, neighboring_cell)))
359 {
360 ++n_neighbors;
361 }
362 }
363 return n_neighbors;
364 }

374 unsigned int
375 n_faces(
376 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell) const
377 {
378 Assert(!is_slave_cell(cell), ExcMessage("You cannot pass a slave cell."));
379 if (is_standard_cell(cell))
380 {
381 return cell->n_faces();
382 }
383 else
384 {
385 const auto & agglomeration = get_agglomerate(cell);
386 unsigned int n_neighbors = 0;
387 for (const auto &cell : agglomeration)
388 {
389 n_neighbors += n_agglomerated_faces_per_cell(cell);
390 }
391 return n_neighbors;
392 }

401 typename DoFHandler<dim, spacedim>::active_cell_iterator
402 agglomerated_neighbor(
403 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
404 const unsigned int f) const
405 {
406 Assert(!is_slave_cell(cell),
407 ExcMessage("This cells is not supposed to be a slave."));
408 if (!at_boundary(cell, f))
409 {
410 if (is_standard_cell(cell) && is_standard_cell(cell->neighbor(f)))
411 {
412 return cell->neighbor(f);
413 }
414 else if (is_master_cell(cell))
415 {
416 const auto &neigh = std::get<1>(master_neighbors[{cell, f}]);
417 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
418 *neigh, &agglo_dh);
419 return cell_dh;
420 }
421 else
422 {
423 // If I fall here, I want to find the neighbor for a standard cell
424 // adjacent to an agglomeration
425 const auto &master_cell =
426 master_slave_relationships_iterators[cell->neighbor(f)
427 ->active_cell_index()];
428 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
429 *master_cell, &agglo_dh);
430 return cell_dh;
431 }
432 }
433 else
434 {
435 return {};
436 }
437 }

446 unsigned int
447 neighbor_of_agglomerated_neighbor(
448 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
449 const unsigned int f) const
450 {
451 Assert(!is_slave_cell(cell),
452 ExcMessage("This cells is not supposed to be a slave."));
453 // First, make sure it's not a boundary face.
454 if (!at_boundary(cell, f))
455 {
456 if (is_standard_cell(cell) && is_standard_cell(cell->neighbor(f)))
457 {
458 return cell->neighbor_of_neighbor(f);
459 }
460 else if (is_master_cell(cell) &&
461 is_master_cell(agglomerated_neighbor(cell, f)))
462 {
463 const auto &current_agglo_info = master_neighbors[{cell, f}];
464 const auto &local_f_idx = std::get<0>(current_agglo_info);
465 const auto &current_cell = std::get<3>(current_agglo_info);
466
467 const auto &agglo_neigh =
468 agglomerated_neighbor(cell,
469 f); // returns the neighboring master
470 const unsigned int n_faces_agglomerated_neighbor =
471 n_faces(agglo_neigh);
472 // Loop over all cells of neighboring agglomerate
473 for (unsigned int f_out = 0; f_out < n_faces_agglomerated_neighbor;
474 ++f_out)
475 {
476 if (agglomerated_neighbor(agglo_neigh, f_out).state() ==
477 IteratorState::valid &&
478 current_cell->neighbor(local_f_idx).state() ==
479 IteratorState::valid &&
480 current_cell.state() == IteratorState::valid)
481 {
482 const auto &neighboring_agglo_info =
483 master_neighbors[{agglo_neigh, f_out}];
484 const auto &local_f_out_idx =
485 std::get<0>(neighboring_agglo_info);
486 const auto &current_cell_out =
487 std::get<3>(neighboring_agglo_info);
488 const auto &other_standard =
489 std::get<1>(neighboring_agglo_info);
490
491 // Here, an extra condition is needed because there can be
492 // more than one face index that returns the same neighbor
493 // if you simply check who is f' s.t.
494 // cell->neigh(f)->neigh(f') == cell. Hence, an extra
495 // condition must be added.
496
497 if (other_standard.state() == IteratorState::valid &&
498 agglomerated_neighbor(agglo_neigh, f_out)
499 ->active_cell_index() ==
500 cell->active_cell_index() &&
501 current_cell->active_cell_index() ==
502 current_cell_out->neighbor(local_f_out_idx)
503 ->active_cell_index() &&
504 current_cell->neighbor(local_f_idx) == current_cell_out)
505 return f_out;
506 }
507 }
508 Assert(false, ExcInternalError());
509 return {}; // just to suppress warnings
510 }
511 else if (is_master_cell(cell) &&
512 is_standard_cell(agglomerated_neighbor(cell, f)))
513 {
514 return std::get<2>(master_neighbors[{cell, f}]);
515 }
516 else
517 {
518 // If I fall here, I want to find the neighbor of neighbor for a
519 // standard cell adjacent to an agglomeration.
520 const auto &master_cell = agglomerated_neighbor(
521 cell, f); // this is the master of the neighboring agglomeration
522
523 return shared_face_agglomeration_idx[{master_cell, cell, f}];
524 }
525 }
526 else
527 {
528 // Face is at boundary
529 return numbers::invalid_unsigned_int;
530 }
531 }

1107 void
1108 setup_master_neighbor_connectivity(
1109 const typename Triangulation<dim, spacedim>::active_cell_iterator
1110 &master_cell) const
1111 {
1112 Assert(master_slave_relationships[master_cell->active_cell_index()] == -1,
1113 ExcMessage("The present cell is not a master one."));
1114
1115 const auto &agglomeration = get_agglomerate(master_cell);
1116
1117 unsigned int n_agglo_faces = 0;
1118 for (const auto &cell : agglomeration)
1119 {
1120 for (const auto f : cell->face_indices())
1121 {
1122 const auto &neighboring_cell = cell->neighbor(f);
1123 // Check if cell is not on the boundary and if it's not
1124 // agglomerated with the neighbor If so, it's a neighbor of the
1125 // present agglomeration
1126 if (neighboring_cell.state() == IteratorState::valid &&
1127 !are_cells_agglomerated(cell, neighboring_cell))
1128 {
1129 // a new face of the agglomeration has been discovered.
1130 polygon_boundary[master_cell].push_back(cell->face(f));
1131
1132 const auto &cell_and_face =
1133 CellAndFace(master_cell, n_agglo_faces); //(agglo,f)
1134 const auto nof = cell->neighbor_of_neighbor(f); // loc(f')
1135
1136 if (is_slave_cell(neighboring_cell))
1137 master_neighbors.emplace(
1138 cell_and_face,
1139 std::make_tuple(f,
1140 master_slave_relationships_iterators
1141 [neighboring_cell->active_cell_index()],
1142 nof,
1143 cell));
1144 else
1145 master_neighbors.emplace(
1146 cell_and_face,
1147 std::make_tuple(
1148 f,
1149 neighboring_cell,
1150 nof,
1151 cell)); //(agglo,f)
1152 //->(loc(f),other_deal_cell,loc(f'),dealcell)
1153
1154 // Now, link the index of the agglomerated face with the
1155 // master and the neighboring cell.
1156 shared_face_agglomeration_idx.emplace(
1157 MasterAndNeighborAndFace(master_cell, neighboring_cell, nof),
1158 n_agglo_faces);
1159 ++n_agglo_faces;
1160 }
1161 else if (cell->face(f)->at_boundary())
1162 {
1163 // Boundary face of a boundary cell.
1164 // Note that the neighboring cell must be invalid.
1165 const auto &cell_and_face =
1166 CellAndFace(master_cell, n_agglo_faces);
1167
1168 polygon_boundary[master_cell].push_back(cell->face(f));
1169
1170 master_neighbors.emplace(
1171 cell_and_face,
1172 std::make_tuple(f,
1173 neighboring_cell,
1174 std::numeric_limits<unsigned int>::max(),
1175 cell)); // TODO: check what the last element
1176 // should be...
1177 ++n_agglo_faces;
1178 }
1179 }
1180 }
1181 }
