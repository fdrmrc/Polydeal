242 template <class FiniteElement>
243 void
244 distribute_agglomerated_dofs(const FiniteElement &fe_space)
245 {
246 Assert((dynamic_cast<const FE_DGQ<dim, spacedim> *>(&fe_space)),
247 ExcNotImplemented(
248 "Currently, this interface supports only DG discretizations."));
249 fe = std::make_unique<FE_DGQ<dim, spacedim>>(fe_space);
250
251 if (hybrid_mesh)
252 {
253 // the mesh is composed by standard and agglomerate cells. initialize
254 // classes needed for standard cells in order to treat that finite
255 // element space as defined on a standard shape and not on the
256 // BoundingBox.
257 standard_scratch =
258 std::make_unique<ScratchData>(*mapping,
259 *fe,
260 QGauss<dim>(2 * fe_space.degree + 2),
261 internal_agglomeration_flags);
262 }
263
264
265 fe_collection.push_back(*fe); // master
266 fe_collection.push_back(FE_Nothing<dim, spacedim>()); // slave
267
268 initialize_hp_structure();
269
270 // in case the tria is distributed, communicate ghost information with
271 // neighboring ranks
272 const bool needs_ghost_info =
273 dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
274 &*tria) != nullptr;
275 if (needs_ghost_info)
276 setup_ghost_polytopes();
277
278 setup_connectivity_of_agglomeration();
279
280 if (needs_ghost_info)
281 exchange_interface_values();
282 }
