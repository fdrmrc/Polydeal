39template <int dim, int spacedim>
40void
41AgglomerationHandler<dim, spacedim>::agglomerate_cells(
42 const std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
43 &vec_of_cells)
44{
45 Assert(master_slave_relationships.size() > 0,
46 ExcMessage("Before calling this function, be sure that the "
47 "constructor of this object has been called."));
48 Assert(vec_of_cells.size() >= 1, ExcMessage("No cells to be agglomerated."));
49
50 // Get global index for each cell
51 std::vector<types::global_cell_index> global_indices;
52 for (const auto &cell : vec_of_cells)
53 global_indices.push_back(cell->active_cell_index());
54
55 // Maximum index drives the selection of the master cell
56 types::global_cell_index master_idx =
57 *std::max_element(global_indices.begin(), global_indices.end());
58
59 for (const types::global_cell_index idx : global_indices)
60 master_slave_relationships[idx] = master_idx; // mark each slave
61
62 for (const auto &cell : vec_of_cells)
63 {
64 if (cell->active_cell_index() == master_idx)
65 master_slave_relationships_iterators[cell->active_cell_index()] =
66 cell; // set iterator to master cell
67 }
68
69 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
70 slaves;
71 slaves.reserve(vec_of_cells.size() - 1);
72 for (const auto &cell : vec_of_cells)
73 {
74 if (cell->active_cell_index() != master_idx)
75 {
76 master_slave_relationships_iterators[cell->active_cell_index()] =
77 master_slave_relationships_iterators[master_idx];
78 slaves.push_back(cell);
79 }
80 }
81 master2slaves[master_idx] = slaves;
82
83 for (const types::global_cell_index idx : global_indices)
84 {
85 master_slave_relationships[idx] = master_idx; // mark each slave
86 }
87
88 master_slave_relationships[master_idx] = -1;
89
90 master2polygon[master_idx] = n_agglomerations;
91 ++n_agglomerations; // agglomeration has been performed, record it
92 create_bounding_box(vec_of_cells, master_idx); // fill the vector of bboxes
93}

108template <int dim, int spacedim>
109Quadrature<dim>
110AgglomerationHandler<dim, spacedim>::agglomerated_quadrature(
111 const std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
112 &cells,
113 const typename Triangulation<dim, spacedim>::active_cell_iterator
114 &master_cell) const
115{
116 Assert(is_master_cell(master_cell),
117 ExcMessage("This must be a master cell."));
118
119 std::vector<Point<dim>> vec_pts;
120 std::vector<double> vec_JxWs;
121 for (const auto &dummy_cell : cells)
122 {
123 no_values->reinit(dummy_cell);
124 auto q_points = no_values->get_quadrature_points(); // real qpoints
125 const auto &JxWs = no_values->get_JxW_values();
126
127 std::transform(q_points.begin(),
128 q_points.end(),
129 std::back_inserter(vec_pts),
130 [&](const Point<spacedim> &p) { return p; });
131 std::transform(JxWs.begin(),
132 JxWs.end(),
133 std::back_inserter(vec_JxWs),
134 [&](const double w) { return w; });
135 }
136
137 // Map back each point in real space by using the map associated to the
138 // bounding box.
139 std::vector<Point<dim>> unit_points(vec_pts.size());
140 const auto & bbox =
141 bboxes[master2polygon.at(master_cell->active_cell_index())];
142 unit_points.reserve(vec_pts.size());
143
144 for (unsigned int i = 0; i < vec_pts.size(); i++)
145 unit_points[i] = bbox.real_to_unit(vec_pts[i]);
146
147 return Quadrature<dim>(unit_points, vec_JxWs);
148}

152template <int dim, int spacedim>
153void
154AgglomerationHandler<dim, spacedim>::initialize_hp_structure()
155{
156 Assert(agglo_dh.get_triangulation().n_cells() > 0,
157 ExcMessage(
158 "Triangulation must not be empty upon calling this function."));
159 // Assert(n_agglomerations > 0,
160 // ExcMessage("No agglomeration has been performed."));
161 for (const auto &cell : agglo_dh.active_cell_iterators())
162 if (is_master_cell(cell))
163 cell->set_active_fe_index(CellAgglomerationType::master);
164 else if (is_slave_cell(cell))
165 cell->set_active_fe_index(CellAgglomerationType::slave); // slave cell
166 else
167 cell->set_active_fe_index(CellAgglomerationType::standard); // standard
168
169 agglo_dh.distribute_dofs(fe_collection);
170 euler_mapping =
171 std::make_unique<MappingFEField<dim, spacedim>>(euler_dh, euler_vector);
172}

176template <int dim, int spacedim>
177const FEValues<dim, spacedim> &
178AgglomerationHandler<dim, spacedim>::reinit(
179 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell) const
180{
181 Assert(euler_mapping,
182 ExcMessage("The mapping describing the physical element stemming from "
183 "agglomeration has not been set up."));
184
185 if (is_master_cell(cell))
186 {
187 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
188 agglo_cells;
189 // Push back that master and slaves
190 agglo_cells.push_back(cell);
191 const auto &slaves = get_slaves_of_idx(cell->active_cell_index());
192 std::transform(
193 slaves.begin(),
194 slaves.end(),
195 std::back_inserter(agglo_cells),
196 [&](const typename Triangulation<dim, spacedim>::active_cell_iterator
197 &c) { return c; });
198
199 Quadrature<dim> agglo_quad = agglomerated_quadrature(agglo_cells, cell);
200
201 const double bbox_measure =
202 bboxes[master2polygon.at(cell->active_cell_index())].volume();
203
204 // Scale weights with the volume of the BBox. This way, the euler_mapping
205 // defining the BBOx doesn't alter them.
206 std::vector<double> scaled_weights;
207 std::transform(agglo_quad.get_weights().begin(),
208 agglo_quad.get_weights().end(),
209 std::back_inserter(scaled_weights),
210 [&bbox_measure](const double w) {
211 return w / bbox_measure;
212 });
213
214 Quadrature<dim> scaled_quad(agglo_quad.get_points(), scaled_weights);
215
216 agglomerated_scratch = std::make_unique<ScratchData>(*euler_mapping,
217 fe_collection[0],
218 scaled_quad,
219 agglomeration_flags);
220 return agglomerated_scratch->reinit(cell);
221 }
222 else if (is_standard_cell(cell))
223 {
224 // ensure the DG space is the same we have from the other DoFHandler(s)
225 standard_scratch = std::make_unique<ScratchData>(*mapping,
226 fe_collection[2],
227 agglomeration_quad,
228 agglomeration_flags);
229 return standard_scratch->reinit(cell);
230 }
231 else
232 {
233 std::vector<Point<dim>> pts{{}};
234 std::vector<double> wgts{0.};
235 Quadrature<dim> dummy_quad(pts, wgts);
236 standard_scratch = std::make_unique<ScratchData>(*mapping,
237 fe_collection[1],
238 dummy_quad,
239 agglomeration_flags);
240 return standard_scratch->reinit(cell);
241 }
242}

246template <int dim, int spacedim>
247const FEValuesBase<dim, spacedim> &
248AgglomerationHandler<dim, spacedim>::reinit_master(
249 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
250 const unsigned int face_number,
251 std::unique_ptr<NonMatching::FEImmersedSurfaceValues<spacedim>>
252 &agglo_isv_ptr) const
253{
254 Assert(is_master_cell(cell), ExcMessage("This cell must be a master one."));
255 const auto &info_neighbors = master_neighbors[{cell, face_number}];
256 const auto &local_face_idx = std::get<0>(info_neighbors);
257 const auto &neighboring_cell = std::get<3>(info_neighbors);
258
259 if (neighboring_cell.state() == IteratorState::valid)
260 {
261 no_face_values->reinit(neighboring_cell, local_face_idx);
262 auto q_points = no_face_values->get_quadrature_points();
263
264 const auto &JxWs = no_face_values->get_JxW_values();
265 const auto &normals = no_face_values->get_normal_vectors();
266
267 const auto & bbox = bboxes[master2polygon.at(cell->active_cell_index())];
268 const double bbox_measure = bbox.volume();
269 std::vector<Point<spacedim>> unit_q_points;
270
271 std::transform(q_points.begin(),
272 q_points.end(),
273 std::back_inserter(unit_q_points),
274 [&](const Point<spacedim> &p) {
275 return bbox.real_to_unit(p);
276 });
277
278 // Weights must be scaled with det(J)*|J^-t n| for each quadrature point.
279 // Use the fact that we are using a BBox, so the jacobi entries are the
280 // side_length in each direction.
281 // Unfortunately, normal vectors will be scaled internally by deal.II by
282 // using a covariant transformation. In order not to change the normals,
283 // we multiply by the correct factors in order to obtain the original
284 // normal after the call to `reinit(cell)`.
285 std::vector<double> scale_factors(q_points.size());
286 std::vector<double> scaled_weights(q_points.size());
287 std::vector<Tensor<1, dim>> scaled_normals(q_points.size());
288
289 for (unsigned int q = 0; q < q_points.size(); ++q)
290 {
291 for (unsigned int direction = 0; direction < spacedim; ++direction)
292 scaled_normals[q][direction] =
293 normals[q][direction] * (bbox.side_length(direction));
294
295 scaled_weights[q] =
296 (JxWs[q] * scaled_normals[q].norm()) / bbox_measure;
297 scaled_normals[q] /= scaled_normals[q].norm();
298 }
299
300 NonMatching::ImmersedSurfaceQuadrature<dim, spacedim> surface_quad(
301 unit_q_points, scaled_weights, scaled_normals);
302
303 agglo_isv_ptr =
304 std::make_unique<NonMatching::FEImmersedSurfaceValues<spacedim>>(
305 *euler_mapping, *fe, surface_quad, agglomeration_face_flags);
306
307 agglo_isv_ptr->reinit(cell);
308
309 return *agglo_isv_ptr;
310 }
311 else
312 {
313 // Then it's a boundary face of an agglomeration living on the
314 // boundary of the tria. You need to return an FEFaceValues on the
315 // boundary face of a boundary cell.
316 no_face_values->reinit(neighboring_cell, local_face_idx);
317
318 // TODO: check if *mapping or *euler_mapping
319 standard_scratch_face_bdary =
320 std::make_unique<ScratchData>(*euler_mapping,
321 fe_collection[2],
322 agglomeration_quad,
323 agglomeration_flags,
324 agglomeration_face_quad,
325 agglomeration_face_flags);
326
327 return standard_scratch_face_bdary->reinit(cell);
328 }
329}

333template <int dim, int spacedim>
334const FEValuesBase<dim, spacedim> &
335AgglomerationHandler<dim, spacedim>::reinit(
336 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
337 const unsigned int face_number) const
338{
339 Assert(euler_mapping,
340 ExcMessage("The mapping describing the physical element stemming from "
341 "agglomeration has not been set up."));
342
343 if ((is_standard_cell(cell) && at_boundary(cell, face_number)) ||
344 (is_standard_cell(cell) &&
345 is_master_cell(agglomerated_neighbor(cell, face_number))))
346 {
347 standard_scratch_face_any =
348 std::make_unique<ScratchData>(*mapping,
349 fe_collection[2],
350 agglomeration_quad,
351 agglomeration_flags,
352 agglomeration_face_quad,
353 agglomeration_face_flags);
354 return standard_scratch_face_any->reinit(cell, face_number);
355 }
356 else
357 {
358 Assert(is_master_cell(cell), ExcMessage("This should be true."));
359 return reinit_master(cell, face_number, agglomerated_isv_bdary);
360 }
361}

365template <int dim, int spacedim>
366std::pair<const FEValuesBase<dim, spacedim> &,
367 const FEValuesBase<dim, spacedim> &>
368AgglomerationHandler<dim, spacedim>::reinit_interface(
369 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell_in,
370 const typename DoFHandler<dim, spacedim>::active_cell_iterator &neigh_cell,
371 const unsigned int local_in,
372 const unsigned int local_neigh) const
373{
374 Assert(
375 !is_slave_cell(cell_in) && !is_slave_cell(neigh_cell),
376 ExcMessage(
377 "At least of the two cells sharing a face is a slave cell. This should never happen if you want to agglomerate some cells together. "));
378
379 if (is_standard_cell(cell_in) && is_standard_cell(neigh_cell))
380 {
381 standard_scratch_face_std =
382 std::make_unique<ScratchData>(*mapping,
383 fe_collection[2],
384 agglomeration_quad,
385 agglomeration_flags,
386 agglomeration_face_quad,
387 agglomeration_face_flags);
388
389 standard_scratch_face_std_neigh =
390 std::make_unique<ScratchData>(*mapping,
391 fe_collection[2],
392 agglomeration_quad,
393 agglomeration_flags,
394 agglomeration_face_quad,
395 agglomeration_face_flags);
396
397 std::pair<const FEValuesBase<dim, spacedim> &,
398 const FEValuesBase<dim, spacedim> &>
399 my_p(standard_scratch_face_std->reinit(cell_in, local_in),
400 standard_scratch_face_std_neigh->reinit(neigh_cell, local_neigh));
401
402 return my_p;
403 }
404 else if (is_standard_cell(neigh_cell) && is_master_cell(cell_in))
405 {
406 const auto &fe_in = reinit(cell_in, local_in);
407 // TODO: check if euler or mapping
408 standard_scratch_face_std_another =
409 std::make_unique<ScratchData>(*mapping,
410 fe_collection[2],
411 agglomeration_quad,
412 agglomeration_flags,
413 agglomeration_face_quad,
414 agglomeration_face_flags);
415
416 std::pair<const FEValuesBase<dim, spacedim> &,
417 const FEValuesBase<dim, spacedim> &>
418 my_p(fe_in,
419 standard_scratch_face_std_another->reinit(neigh_cell,
420 local_neigh));
421 return my_p;
422 }
423 else if (is_standard_cell(cell_in) && is_master_cell(neigh_cell))
424 {
425 const auto &fe_out = reinit(neigh_cell, local_neigh);
426 standard_scratch_face_std_another =
427 std::make_unique<ScratchData>(*mapping,
428 fe_collection[2],
429 agglomeration_quad,
430 agglomeration_flags,
431 agglomeration_face_quad,
432 agglomeration_face_flags);
433
434 std::pair<const FEValuesBase<dim, spacedim> &,
435 const FEValuesBase<dim, spacedim> &>
436 my_p(standard_scratch_face_std_another->reinit(cell_in, local_in),
437 fe_out);
438
439 return my_p;
440 }
441 else
442 {
443 Assert(is_master_cell(cell_in) && is_master_cell(neigh_cell),
444 ExcMessage("Both cells should be masters."));
445 // both are masters. That means you want to compute the jumps or
446 // averages between a face shared by two neighboring agglomerations.
447 // This feature is not implemented yet
448
449 const auto &fe_in = reinit_master(cell_in, local_in, agglomerated_isv);
450 const auto &fe_out =
451 reinit_master(neigh_cell, local_neigh, agglomerated_isv_neigh);
452 std::pair<const FEValuesBase<dim, spacedim> &,
453 const FEValuesBase<dim, spacedim> &>
454 my_p(fe_in, fe_out);
455
456 return my_p;
457 }
458}

462template <int dim, int spacedim>
463void
464AgglomerationHandler<dim, spacedim>::create_agglomeration_sparsity_pattern(
465 SparsityPattern &sparsity_pattern)
466{
467 // Assert(n_agglomerations > 0,
468 // ExcMessage("The agglomeration has not been set up correctly."));
469 Assert(sparsity_pattern.empty(),
470 ExcMessage(
471 "The Sparsity pattern must be empty upon calling this function."));
472
473 DynamicSparsityPattern dsp(agglo_dh.n_dofs(), agglo_dh.n_dofs());
474 AffineConstraints<double> constraints;
475 const bool keep_constrained_dofs = true;
476 // The following lambda is used to teach to `make_flux_sparsity_pattern()`
477 // to couple only cells that are standard, not also slaves and master cells,
478 // for which we need to compute DoFs separately later.
479
480 const auto face_has_flux_coupling =
481 [&](const auto &cell, const types::global_cell_index face_index) {
482 return master_slave_relationships[cell->active_cell_index()] *
483 master_slave_relationships[cell->neighbor(face_index)
484 ->active_cell_index()] ==
485 +4;
486 };
487 const unsigned int n_components = fe_collection.n_components();
488 Table<2, DoFTools::Coupling> cell_coupling(n_components, n_components);
489 Table<2, DoFTools::Coupling> face_coupling(n_components, n_components);
490 cell_coupling[0][0] = DoFTools::always;
491 face_coupling[0][0] = DoFTools::always;
492 DoFTools::make_flux_sparsity_pattern(agglo_dh,
493 dsp,
494 constraints,
495 keep_constrained_dofs,
496 cell_coupling,
497 face_coupling,
498 numbers::invalid_subdomain_id,
499 face_has_flux_coupling);
500
501
502 std::vector<types::global_dof_index> dof_indices_master(
503 agglo_dh.get_fe(0).n_dofs_per_cell());
504 std::vector<types::global_dof_index> dof_indices_neighbor(
505 agglo_dh.get_fe(2).n_dofs_per_cell());
506
507 // Get the information about the neighboring
508 // cells and add the corresponding entries to the sparsity pattern.
509
510 for (const auto &value : master_neighbors)
511 {
512 // value.first is a pair storing master_cell and agglo_face index.
513 // Another `first` is necessary to get the cell.
514 const auto &master_cell = (value.first).first;
515 typename DoFHandler<dim, spacedim>::cell_iterator master_cell_dh(
516 *master_cell, &agglo_dh);
517 master_cell_dh->get_dof_indices(dof_indices_master);
518
519 const auto &neigh = std::get<1>(value.second);
520 if (neigh.state() == IteratorState::valid)
521 {
522 typename DoFHandler<dim, spacedim>::cell_iterator cell_slave(
523 *neigh, &agglo_dh);
524 cell_slave->get_dof_indices(dof_indices_neighbor);
525 for (const unsigned int row_idx : dof_indices_master)
526 dsp.add_entries(row_idx,
527 dof_indices_neighbor.begin(),
528 dof_indices_neighbor.end());
529 for (const unsigned int col_idx : dof_indices_neighbor)
530 dsp.add_entries(col_idx,
531 dof_indices_master.begin(),
532 dof_indices_master.end());
533 }
534 }
535
536
537 sparsity_pattern.copy_from(dsp);
538}

541template <int dim, int spacedim>
542
543void
544AgglomerationHandler<dim, spacedim>::setup_output_interpolation_matrix()
545{
546 Assert(fe->has_generalized_support_points(),
547 ExcMessage("The present FiniteElement is not interpolatory."));
548
549 // Setup an auxiliary DoFHandler for output purposes
550 output_dh.reinit(*tria);
551 output_dh.distribute_dofs(*fe);
552
553 DynamicSparsityPattern dsp(output_dh.n_dofs(), agglo_dh.n_dofs());
554
555 std::vector<types::global_dof_index> agglo_dof_indices(fe->dofs_per_cell);
556 std::vector<types::global_dof_index> standard_dof_indices(fe->dofs_per_cell);
557 std::vector<types::global_dof_index> output_dof_indices(fe->dofs_per_cell);
558
559 Quadrature<dim> quad(fe->get_unit_support_points());
560 FEValues<dim, spacedim> output_fe_values(*fe, quad, update_quadrature_points);
561
562 for (const auto &cell : agglo_dh.active_cell_iterators())
563 if (is_master_cell(cell))
564 {
565 auto slaves = get_slaves_of_idx(cell->active_cell_index());
566 slaves.emplace_back(cell);
567
568 cell->get_dof_indices(agglo_dof_indices);
569
570 for (const auto &slave : slaves)
571 {
572 // addd master-slave relationship
573 const auto slave_output = slave->as_dof_handler_iterator(output_dh);
574 slave_output->get_dof_indices(output_dof_indices);
575 for (const auto row : output_dof_indices)
576 dsp.add_entries(row,
577 agglo_dof_indices.begin(),
578 agglo_dof_indices.end());
579 }
580 }
581 else if (is_standard_cell(cell))
582 {
583 cell->get_dof_indices(agglo_dof_indices);
584 const auto standard_output = cell->as_dof_handler_iterator(output_dh);
585 standard_output->get_dof_indices(standard_dof_indices);
586 for (const auto row : standard_dof_indices)
587 dsp.add_entries(row,
588 agglo_dof_indices.begin(),
589 agglo_dof_indices.end());
590 }
591 output_interpolation_sparsity.copy_from(dsp);
592 output_interpolation_matrix.reinit(output_interpolation_sparsity);
593
594 FullMatrix<double> local_matrix(fe->dofs_per_cell, fe->dofs_per_cell);
595 std::vector<Point<dim>> reference_q_points(fe->dofs_per_cell);
596
597 // Dummy DoFHandler, only needed for loc2glb
598 AffineConstraints<double> c;
599 c.close();
600
601 for (const auto &cell : agglo_dh.active_cell_iterators())
602 {
603 if (is_master_cell(cell))
604 {
605 auto slaves = get_slaves_of_idx(cell->active_cell_index());
606 slaves.emplace_back(cell);
607
608 cell->get_dof_indices(agglo_dof_indices);
609
610 for (const auto &slave : slaves)
611 {
612 // addd master-slave relationship
613 const auto slave_output =
614 slave->as_dof_handler_iterator(output_dh);
615
616 slave_output->get_dof_indices(output_dof_indices);
617 output_fe_values.reinit(slave_output);
618
619 local_matrix = 0.;
620
621 const auto &q_points = output_fe_values.get_quadrature_points();
622 for (const auto i : output_fe_values.dof_indices())
623 {
624 const auto &p =
625 bboxes[master2polygon.at(cell->active_cell_index())]
626 .real_to_unit(q_points[i]);
627 for (const auto j : output_fe_values.dof_indices())
628 {
629 local_matrix(i, j) = fe->shape_value(j, p);
630 }
631 }
632 c.distribute_local_to_global(local_matrix,
633 output_dof_indices,
634 agglo_dof_indices,
635 output_interpolation_matrix);
636 }
637 }
638 else if (is_standard_cell(cell))
639 {
