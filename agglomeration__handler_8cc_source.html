39template <int dim, int spacedim>
40typename AgglomerationHandler<dim, spacedim>::agglomeration_iterator
41AgglomerationHandler<dim, spacedim>::define_agglomerate(
42 const AgglomerationContainer &cells)
43{
44 Assert(master_slave_relationships.size() > 0,
45 ExcMessage("Before calling this function, be sure that the "
46 "constructor of this object has been called."));
47 Assert(cells.size() > 0, ExcMessage("No cells to be agglomerated."));
48
49
50 // First index drives the selection of the master cell. After that, store the
51 // master cell.
52 const types::global_cell_index master_idx = cells[0]->active_cell_index();
53 master_cells_container.push_back(cells[0]);
54 master_slave_relationships[master_idx] = -1;
55
56 // Store slave cells and save the relationship with the parent
57 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
58 slaves;
59 slaves.reserve(cells.size() - 1);
60 // exclude first cell since it's the master cell
61 for (auto it = ++cells.begin(); it != cells.end(); ++it)
62 {
63 slaves.push_back(*it);
64 master_slave_relationships[(*it)->active_cell_index()] =
65 master_idx; // mark each slave
66 master_slave_relationships_iterators[(*it)->active_cell_index()] =
67 cells[0];
68 }
69
70 master_slave_relationships_iterators[master_idx] =
71 cells[0]; // set iterator to master cell
72
73 // Store the slaves of each master
74 master2slaves[master_idx] = slaves;
75 // Save to which polygon this agglomerate correspond
76 master2polygon[master_idx] = n_agglomerations;
77
78 ++n_agglomerations; // an agglomeration has been performed, record it
79
80 create_bounding_box(cells, master_idx); // fill the vector of bboxes
81
82 // Finally, return a polygonal iterator to the polytope just constructed.
83 return {cells[0], this};
84}

87template <int dim, int spacedim>
88Quadrature<dim>
89AgglomerationHandler<dim, spacedim>::agglomerated_quadrature(
90 const typename AgglomerationHandler<dim, spacedim>::AgglomerationContainer
91 &cells,
92 const typename Triangulation<dim, spacedim>::active_cell_iterator
93 &master_cell) const
94{
95 Assert(is_master_cell(master_cell),
96 ExcMessage("This must be a master cell."));
97
98 std::vector<Point<dim>> vec_pts;
99 std::vector<double> vec_JxWs;
100 for (const auto &dummy_cell : cells)
101 {
102 no_values->reinit(dummy_cell);
103 auto q_points = no_values->get_quadrature_points(); // real qpoints
104 const auto &JxWs = no_values->get_JxW_values();
105
106 std::transform(q_points.begin(),
107 q_points.end(),
108 std::back_inserter(vec_pts),
109 [&](const Point<spacedim> &p) { return p; });
110 std::transform(JxWs.begin(),
111 JxWs.end(),
112 std::back_inserter(vec_JxWs),
113 [&](const double w) { return w; });
114 }
115
116 // Map back each point in real space by using the map associated to the
117 // bounding box.
118 std::vector<Point<dim>> unit_points(vec_pts.size());
119 const auto & bbox =
120 bboxes[master2polygon.at(master_cell->active_cell_index())];
121 unit_points.reserve(vec_pts.size());
122
123 for (unsigned int i = 0; i < vec_pts.size(); i++)
124 unit_points[i] = bbox.real_to_unit(vec_pts[i]);
125
126 return Quadrature<dim>(unit_points, vec_JxWs);
127}

131template <int dim, int spacedim>
132void
133AgglomerationHandler<dim, spacedim>::initialize_hp_structure()
134{
135 Assert(agglo_dh.get_triangulation().n_cells() > 0,
136 ExcMessage(
137 "Triangulation must not be empty upon calling this function."));
138 // Assert(n_agglomerations > 0,
139 // ExcMessage("No agglomeration has been performed."));
140 for (const auto &cell : agglo_dh.active_cell_iterators())
141 if (is_master_cell(cell))
142 cell->set_active_fe_index(CellAgglomerationType::master);
143 else if (is_slave_cell(cell))
144 cell->set_active_fe_index(CellAgglomerationType::slave); // slave cell
145 else
146 cell->set_active_fe_index(CellAgglomerationType::standard); // standard
147
148 agglo_dh.distribute_dofs(fe_collection);
149 euler_mapping =
150 std::make_unique<MappingFEField<dim, spacedim>>(euler_dh, euler_vector);
151}

155template <int dim, int spacedim>
156const FEValues<dim, spacedim> &
157AgglomerationHandler<dim, spacedim>::reinit(
158 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell) const
159{
160 Assert(euler_mapping,
161 ExcMessage("The mapping describing the physical element stemming from "
162 "agglomeration has not been set up."));
163
164 if (is_master_cell(cell))
165 {
166 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
167 agglo_cells;
168 // Push back that master and slaves
169 agglo_cells.push_back(cell);
170 const auto &slaves = get_slaves_of_idx(cell->active_cell_index());
171 std::transform(
172 slaves.begin(),
173 slaves.end(),
174 std::back_inserter(agglo_cells),
175 [&](const typename Triangulation<dim, spacedim>::active_cell_iterator
176 &c) { return c; });
177
178 Quadrature<dim> agglo_quad = agglomerated_quadrature(agglo_cells, cell);
179
180 const double bbox_measure =
181 bboxes[master2polygon.at(cell->active_cell_index())].volume();
182
183 // Scale weights with the volume of the BBox. This way, the euler_mapping
184 // defining the BBOx doesn't alter them.
185 std::vector<double> scaled_weights;
186 std::transform(agglo_quad.get_weights().begin(),
187 agglo_quad.get_weights().end(),
188 std::back_inserter(scaled_weights),
189 [&bbox_measure](const double w) {
190 return w / bbox_measure;
191 });
192
193 Quadrature<dim> scaled_quad(agglo_quad.get_points(), scaled_weights);
194
195 agglomerated_scratch = std::make_unique<ScratchData>(*euler_mapping,
196 fe_collection[0],
197 scaled_quad,
198 agglomeration_flags);
199 return agglomerated_scratch->reinit(cell);
200 }
201 else if (is_standard_cell(cell))
202 {
203 // ensure the DG space is the same we have from the other DoFHandler(s)
204 standard_scratch = std::make_unique<ScratchData>(*mapping,
205 fe_collection[2],
206 agglomeration_quad,
207 agglomeration_flags);
208 return standard_scratch->reinit(cell);
209 }
210 else
211 {
212 std::vector<Point<dim>> pts{{}};
213 std::vector<double> wgts{0.};
214 Quadrature<dim> dummy_quad(pts, wgts);
215 standard_scratch = std::make_unique<ScratchData>(*mapping,
216 fe_collection[1],
217 dummy_quad,
218 agglomeration_flags);
219 return standard_scratch->reinit(cell);
220 }
221}

368template <int dim, int spacedim>
369std::pair<const FEValuesBase<dim, spacedim> &,
370 const FEValuesBase<dim, spacedim> &>
371AgglomerationHandler<dim, spacedim>::reinit_interface(
372 const AgglomerationIterator<dim, spacedim> &polytope_in,
373 const AgglomerationIterator<dim, spacedim> &neigh_polytope,
374 const unsigned int local_in,
375 const unsigned int local_neigh) const
376{
377 const auto &cell_in = polytope_in->as_dof_handler_iterator(agglo_dh);
378 const auto &neigh_cell = neigh_polytope->as_dof_handler_iterator(agglo_dh);
379 Assert(
380 !is_slave_cell(cell_in) && !is_slave_cell(neigh_cell),
381 ExcMessage(
382 "At least of the two cells sharing a face is a slave cell. This should never happen if you want to agglomerate some cells together. "));
383
384 if (is_standard_cell(cell_in) && is_standard_cell(neigh_cell))
385 {
386 standard_scratch_face_std =
387 std::make_unique<ScratchData>(*mapping,
388 fe_collection[2],
389 agglomeration_quad,
390 agglomeration_flags,
391 agglomeration_face_quad,
392 agglomeration_face_flags);
393
394 standard_scratch_face_std_neigh =
395 std::make_unique<ScratchData>(*mapping,
396 fe_collection[2],
397 agglomeration_quad,
398 agglomeration_flags,
399 agglomeration_face_quad,
400 agglomeration_face_flags);
401
402 std::pair<const FEValuesBase<dim, spacedim> &,
403 const FEValuesBase<dim, spacedim> &>
404 my_p(standard_scratch_face_std->reinit(cell_in, local_in),
405 standard_scratch_face_std_neigh->reinit(neigh_cell, local_neigh));
406
407 return my_p;
408 }
409 else if (is_standard_cell(neigh_cell) && is_master_cell(cell_in))
410 {
411 const auto &fe_in = reinit(cell_in, local_in);
412 // TODO: check if euler or mapping
413 standard_scratch_face_std_another =
414 std::make_unique<ScratchData>(*mapping,
415 fe_collection[2],
416 agglomeration_quad,
417 agglomeration_flags,
418 agglomeration_face_quad,
419 agglomeration_face_flags);
420
421 std::pair<const FEValuesBase<dim, spacedim> &,
422 const FEValuesBase<dim, spacedim> &>
423 my_p(fe_in,
424 standard_scratch_face_std_another->reinit(neigh_cell,
425 local_neigh));
426 return my_p;
427 }
428 else if (is_standard_cell(cell_in) && is_master_cell(neigh_cell))
429 {
430 const auto &fe_out = reinit(neigh_cell, local_neigh);
431 standard_scratch_face_std_another =
432 std::make_unique<ScratchData>(*mapping,
433 fe_collection[2],
434 agglomeration_quad,
435 agglomeration_flags,
436 agglomeration_face_quad,
437 agglomeration_face_flags);
438
439 std::pair<const FEValuesBase<dim, spacedim> &,
440 const FEValuesBase<dim, spacedim> &>
441 my_p(standard_scratch_face_std_another->reinit(cell_in, local_in),
442 fe_out);
443
444 return my_p;
445 }
446 else
447 {
448 Assert(is_master_cell(cell_in) && is_master_cell(neigh_cell),
449 ExcMessage("Both cells should be masters."));
450 // both are masters. That means you want to compute the jumps or
451 // averages between a face shared by two neighboring agglomerations.
452
453 const auto &fe_in =
454 internal::AgglomerationHandlerImplementation<dim, spacedim>::
455 reinit_master(cell_in, local_in, agglomerated_isv, *this);
456 const auto &fe_out =
457 internal::AgglomerationHandlerImplementation<dim, spacedim>::
458 reinit_master(neigh_cell, local_neigh, agglomerated_isv_neigh, *this);
459 std::pair<const FEValuesBase<dim, spacedim> &,
460 const FEValuesBase<dim, spacedim> &>
461 my_p(fe_in, fe_out);
462
463 return my_p;
464 }

568template <int dim, int spacedim>
569void
570AgglomerationHandler<dim, spacedim>::create_agglomeration_sparsity_pattern(
571 SparsityPattern &sparsity_pattern)
572{
573 // Assert(n_agglomerations > 0,
574 // ExcMessage("The agglomeration has not been set up correctly."));
575 Assert(sparsity_pattern.empty(),
576 ExcMessage(
577 "The Sparsity pattern must be empty upon calling this function."));
578
579 DynamicSparsityPattern dsp(agglo_dh.n_dofs(), agglo_dh.n_dofs());
580 AffineConstraints<double> constraints;
581 const bool keep_constrained_dofs = true;
582 // The following lambda is used to teach to `make_flux_sparsity_pattern()`
583 // to couple only cells that are standard, not also slaves and master cells,
584 // for which we need to compute DoFs separately later.
585
586 const auto face_has_flux_coupling =
587 [&](const auto &cell, const types::global_cell_index face_index) {
588 return master_slave_relationships[cell->active_cell_index()] *
589 master_slave_relationships[cell->neighbor(face_index)
590 ->active_cell_index()] ==
591 +4;
592 };
593 const unsigned int n_components = fe_collection.n_components();
594 Table<2, DoFTools::Coupling> cell_coupling(n_components, n_components);
595 Table<2, DoFTools::Coupling> face_coupling(n_components, n_components);
596 cell_coupling[0][0] = DoFTools::always;
597 face_coupling[0][0] = DoFTools::always;
598 DoFTools::make_flux_sparsity_pattern(agglo_dh,
599 dsp,
600 constraints,
601 keep_constrained_dofs,
602 cell_coupling,
603 face_coupling,
604 numbers::invalid_subdomain_id,
605 face_has_flux_coupling);
606
607
608 std::vector<types::global_dof_index> dof_indices_master(
609 agglo_dh.get_fe(0).n_dofs_per_cell());
610 std::vector<types::global_dof_index> dof_indices_neighbor(
611 agglo_dh.get_fe(2).n_dofs_per_cell());
612
613 // Get the information about the neighboring
614 // cells and add the corresponding entries to the sparsity pattern.
615
616 for (const auto &value : master_neighbors)
617 {
618 // value.first is a pair storing master_cell and agglo_face index.
619 // Another `first` is necessary to get the cell.
620 const auto &master_cell = (value.first).first;
621 typename DoFHandler<dim, spacedim>::cell_iterator master_cell_dh(
622 *master_cell, &agglo_dh);
623 master_cell_dh->get_dof_indices(dof_indices_master);
624
625 const auto &neigh = std::get<1>(value.second);
626 if (neigh.state() == IteratorState::valid)
627 {
628 typename DoFHandler<dim, spacedim>::cell_iterator cell_slave(
629 *neigh, &agglo_dh);
630 cell_slave->get_dof_indices(dof_indices_neighbor);
631 for (const unsigned int row_idx : dof_indices_master)
632 dsp.add_entries(row_idx,
633 dof_indices_neighbor.begin(),
634 dof_indices_neighbor.end());
635 for (const unsigned int col_idx : dof_indices_neighbor)
636 dsp.add_entries(col_idx,
637 dof_indices_master.begin(),
638 dof_indices_master.end());
639 }
640 }
641
642
643 sparsity_pattern.copy_from(dsp);
644}

649void
650AgglomerationHandler<dim, spacedim>::setup_output_interpolation_matrix()
651{
652 Assert(fe->has_generalized_support_points(),
653 ExcMessage("The present FiniteElement is not interpolatory."));
654
655 // Setup an auxiliary DoFHandler for output purposes
656 output_dh.reinit(*tria);
657 output_dh.distribute_dofs(*fe);
658
659 DynamicSparsityPattern dsp(output_dh.n_dofs(), agglo_dh.n_dofs());
660
661 std::vector<types::global_dof_index> agglo_dof_indices(fe->dofs_per_cell);
662 std::vector<types::global_dof_index> standard_dof_indices(fe->dofs_per_cell);
663 std::vector<types::global_dof_index> output_dof_indices(fe->dofs_per_cell);
664
665 Quadrature<dim> quad(fe->get_unit_support_points());
666 FEValues<dim, spacedim> output_fe_values(*fe, quad, update_quadrature_points);
667
668 for (const auto &cell : agglo_dh.active_cell_iterators())
669 if (is_master_cell(cell))
670 {
671 auto slaves = get_slaves_of_idx(cell->active_cell_index());
672 slaves.emplace_back(cell);
673
674 cell->get_dof_indices(agglo_dof_indices);
675
676 for (const auto &slave : slaves)
677 {
678 // addd master-slave relationship
679 const auto slave_output = slave->as_dof_handler_iterator(output_dh);
680 slave_output->get_dof_indices(output_dof_indices);
681 for (const auto row : output_dof_indices)
682 dsp.add_entries(row,
683 agglo_dof_indices.begin(),
684 agglo_dof_indices.end());
685 }
686 }
687 else if (is_standard_cell(cell))
688 {
689 cell->get_dof_indices(agglo_dof_indices);
690 const auto standard_output = cell->as_dof_handler_iterator(output_dh);
691 standard_output->get_dof_indices(standard_dof_indices);
692 for (const auto row : standard_dof_indices)
693 dsp.add_entries(row,
694 agglo_dof_indices.begin(),
695 agglo_dof_indices.end());
696 }
697 output_interpolation_sparsity.copy_from(dsp);
698 output_interpolation_matrix.reinit(output_interpolation_sparsity);
699
700 FullMatrix<double> local_matrix(fe->dofs_per_cell, fe->dofs_per_cell);
701 std::vector<Point<dim>> reference_q_points(fe->dofs_per_cell);
702
703 // Dummy DoFHandler, only needed for loc2glb
704 AffineConstraints<double> c;
705 c.close();
706
707 for (const auto &cell : agglo_dh.active_cell_iterators())
708 {
709 if (is_master_cell(cell))
710 {
711 auto slaves = get_slaves_of_idx(cell->active_cell_index());
712 slaves.emplace_back(cell);
713
714 cell->get_dof_indices(agglo_dof_indices);
715
716 for (const auto &slave : slaves)
717 {
718 // addd master-slave relationship
719 const auto slave_output =
720 slave->as_dof_handler_iterator(output_dh);
721
722 slave_output->get_dof_indices(output_dof_indices);
723 output_fe_values.reinit(slave_output);
724
725 local_matrix = 0.;
726
727 const auto &q_points = output_fe_values.get_quadrature_points();
728 for (const auto i : output_fe_values.dof_indices())
729 {
730 const auto &p =
731 bboxes[master2polygon.at(cell->active_cell_index())]
732 .real_to_unit(q_points[i]);
733 for (const auto j : output_fe_values.dof_indices())
734 {
735 local_matrix(i, j) = fe->shape_value(j, p);
736 }
737 }
738 c.distribute_local_to_global(local_matrix,
739 output_dof_indices,
740 agglo_dof_indices,
741 output_interpolation_matrix);
742 }
743 }
744 else if (is_standard_cell(cell))
745 {
746 cell->get_dof_indices(agglo_dof_indices);
747
748 const auto standard_output = cell->as_dof_handler_iterator(output_dh);
749
750 standard_output->get_dof_indices(standard_dof_indices);
751 output_fe_values.reinit(standard_output);
752
753 local_matrix = 0.;
754 for (const auto i : output_fe_values.dof_indices())
755 local_matrix(i, i) = 1.;
756 c.distribute_local_to_global(local_matrix,
757 standard_dof_indices,
758 agglo_dof_indices,
759 output_interpolation_matrix);
760 }
761 }
762}
