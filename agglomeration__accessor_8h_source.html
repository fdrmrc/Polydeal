227template <int dim, int spacedim>
228unsigned int
229AgglomerationAccessor<dim, spacedim>::n_agglomerated_faces_per_cell(
230 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell) const
231{
232 unsigned int n_neighbors = 0;
233 for (const auto &f : cell->face_indices())
234 {
235 const auto &neighboring_cell = cell->neighbor(f);
236 if ((cell->face(f)->at_boundary()) ||
237 (neighboring_cell->is_active() &&
238 !handler->are_cells_agglomerated(cell, neighboring_cell)))
239 {
240 ++n_neighbors;
241 }
242 }
243 return n_neighbors;
244}

247template <int dim, int spacedim>
248unsigned int
249AgglomerationAccessor<dim, spacedim>::n_faces() const
250{
251 Assert(!handler->is_slave_cell(master_cell),
252 ExcMessage("You cannot pass a slave cell."));
253 if (handler->is_standard_cell(master_cell))
254 {
255 return master_cell->n_faces();
256 }
257 else
258 {
259 const auto & agglomeration = get_agglomerate();
260 unsigned int n_neighbors = 0;
261 for (const auto &cell : agglomeration)
262 {
263 n_neighbors += n_agglomerated_faces_per_cell(cell);
264 }
265 return n_neighbors;
266 }
267}

271template <int dim, int spacedim>
272const AgglomerationIterator<dim, spacedim>
273AgglomerationAccessor<dim, spacedim>::neighbor(const unsigned int f) const
274{
275 // Assert(!handler->is_slave_cell(cell),
276 // ExcMessage("This cells is not supposed to be a slave."));
277 if (!at_boundary(f))
278 {
279 if (handler->is_standard_cell(master_cell) &&
280 handler->is_standard_cell(master_cell->neighbor(f)))
281 {
282 // return master_cell->neighbor(f);
283 return {master_cell->neighbor(f), handler};
284 }
285 else if (handler->is_master_cell(master_cell))
286 {
287 const auto &neigh =
288 std::get<1>(handler->master_neighbors[{master_cell, f}]);
289 // typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
290 // *neigh, &(handler->agglo_dh));
291 // return cell_dh;
292 return {neigh, handler};
293 }
294 else
295 {
296 // If I fall here, I want to find the neighbor for a standard cell
297 // adjacent to an agglomeration
298 const auto &cell = handler->master_slave_relationships_iterators
299 [master_cell->neighbor(f)->active_cell_index()];
300 // typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
301 // *master_cell, &(handler->agglo_dh));
302 // return cell_dh;
303 return {cell, handler};
304 }
305 }
306 else
307 {
308 return {};
309 }
310}

314template <int dim, int spacedim>
315unsigned int
316AgglomerationAccessor<dim, spacedim>::neighbor_of_agglomerated_neighbor(
317 const unsigned int f) const
318{
319 // Assert(!is_slave_cell(cell),
320 // ExcMessage("This cells is not supposed to be a slave."));
321 // First, make sure it's not a boundary face.
322 if (!at_boundary(f))
323 {
324 if (handler->is_standard_cell(master_cell) &&
325 handler->is_standard_cell(master_cell->neighbor(f)))
326 {
327 return master_cell->neighbor_of_neighbor(f);
328 }
329 else if (handler->is_master_cell(master_cell) &&
330 handler->is_master_cell(neighbor(f).master_cell()))
331 {
332 const auto &current_agglo_info =
333 handler->master_neighbors[{master_cell, f}];
334 const auto &local_f_idx = std::get<0>(current_agglo_info);
335 const auto &current_cell = std::get<3>(current_agglo_info);
336
337 const auto &agglo_neigh =
338 neighbor(f); // returns the neighboring master
339 const unsigned int n_faces_agglomerated_neighbor =
340 agglo_neigh->n_faces();
341
342 // Loop over all cells of neighboring agglomerate
343 for (unsigned int f_out = 0; f_out < n_faces_agglomerated_neighbor;
344 ++f_out)
345 {
346 if (agglo_neigh->neighbor(f_out).state() ==
347 IteratorState::valid &&
348 current_cell->neighbor(local_f_idx).state() ==
349 IteratorState::valid &&
350 current_cell.state() == IteratorState::valid)
351 {
352 const auto &neighboring_agglo_info =
353 handler
354 ->master_neighbors[{agglo_neigh.master_cell(), f_out}];
355 const auto &local_f_out_idx =
356 std::get<0>(neighboring_agglo_info);
357 const auto &current_cell_out =
358 std::get<3>(neighboring_agglo_info);
359 const auto &other_standard =
360 std::get<1>(neighboring_agglo_info);
361
362 // Here, an extra condition is needed because there can be
363 // more than one face index that returns the same neighbor
364 // if you simply check who is f' s.t.
365 // cell->neigh(f)->neigh(f') == cell. Hence, an extra
366 // condition must be added.
367
368 if (other_standard.state() == IteratorState::valid &&
369 agglo_neigh->neighbor(f_out)->index() == index() &&
370 current_cell->active_cell_index() ==
371 current_cell_out->neighbor(local_f_out_idx)
372 ->active_cell_index() &&
373 current_cell->neighbor(local_f_idx) == current_cell_out)
374 return f_out;
375 }
376 }
377 Assert(false, ExcInternalError());
378 return {}; // just to suppress warnings
379 }
380 else if (handler->is_master_cell(master_cell) &&
381 handler->is_standard_cell(neighbor(f).master_cell()))
382 {
383 return std::get<2>(handler->master_neighbors[{master_cell, f}]);
384 }
385 else
386 {
387 // TODO : check
388 // If I fall here, I want to find the neighbor of neighbor for a
389 // standard cell adjacent to an agglomeration.
390 const auto &master_neigh =
391 neighbor(f); // this is the master of the neighboring agglomeration
392 return handler->shared_face_agglomeration_idx[{
393 master_neigh.master_cell(), master_cell, f}];
394 }
395 }
396 else
397 {
398 // Face is at boundary
399 return numbers::invalid_unsigned_int;
400 }
401}

411template <int dim, int spacedim>
412inline AgglomerationAccessor<dim, spacedim>::AgglomerationAccessor(
413 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell,
414 const AgglomerationHandler<dim, spacedim> * ah)
415{
416 handler = const_cast<AgglomerationHandler<dim, spacedim> *>(ah);
417 if (&(*handler->master_cells_container.end()) == std::addressof(cell))
418 {
419 present_index = handler->master_cells_container.size();
420 master_cell = handler->master_cells_container[present_index];
421 }
422 else
423 {
424 present_index = handler->master2polygon.at(cell->active_cell_index());
425 master_cell = cell;
426 }
427}

458template <int dim, int spacedim>
459bool
460AgglomerationAccessor<dim, spacedim>::at_boundary(const unsigned int f) const
461{
462 Assert(!handler->is_slave_cell(master_cell),
463 ExcMessage("This function should not be called for a slave cell."));
464 if (handler->is_standard_cell(master_cell))
465 return master_cell->face(f)->at_boundary();
466 else
467 return std::get<2>(handler->master_neighbors[{master_cell, f}]) ==
468 std::numeric_limits<unsigned int>::max();
469}

482template <int dim, int spacedim>
483inline double
484AgglomerationAccessor<dim, spacedim>::diameter() const
485{
486 Assert(!handler->is_slave_cell(master_cell),
487 ExcMessage("The present function cannot be called for slave cells."));
488
489 if (handler->is_master_cell(master_cell))
490 {
491 // Get the bounding box associated with the master cell
492 const auto &bdary_pts =
493 handler->bboxes[present_index].get_boundary_points();
494 return (bdary_pts.second - bdary_pts.first).norm();
495 }
496 else
497 {
498 // Standard deal.II way to get the measure of a cell.
499 return master_cell->diameter();
500 }
501}
