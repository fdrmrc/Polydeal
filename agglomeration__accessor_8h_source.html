237template <int dim, int spacedim>
238unsigned int
239AgglomerationAccessor<dim, spacedim>::n_agglomerated_faces_per_cell(
240 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell) const
241{
242 unsigned int n_neighbors = 0;
243 for (const auto &f : cell->face_indices())
244 {
245 const auto &neighboring_cell = cell->neighbor(f);
246 if ((cell->face(f)->at_boundary()) ||
247 (neighboring_cell->is_active() &&
248 !handler->are_cells_agglomerated(cell, neighboring_cell)))
249 {
250 ++n_neighbors;
251 }
252 }
253 return n_neighbors;
254}

269template <int dim, int spacedim>
270const AgglomerationIterator<dim, spacedim>
271AgglomerationAccessor<dim, spacedim>::neighbor(const unsigned int f) const
272{
273 if (!at_boundary(f))
274 {
275 const auto &neigh =
276 handler->polytope_cache.cell_face_at_boundary
277 .at({handler->master2polygon.at(master_cell->active_cell_index()), f})
278 .second;
279 typename DoFHandler<dim, spacedim>::active_cell_iterator cell_dh(
280 *neigh, &(handler->agglo_dh));
281 return {cell_dh, handler};
282 }
283 else
284 {
285 return {};
286 }
287}

291template <int dim, int spacedim>
292unsigned int
293AgglomerationAccessor<dim, spacedim>::neighbor_of_agglomerated_neighbor(
294 const unsigned int f) const
295{
296 // First, make sure it's not a boundary face.
297 if (!at_boundary(f))
298 {
299 const auto &neigh_polytope =
300 neighbor(f); // returns the neighboring master
301
302 AssertThrow(neigh_polytope.state() == IteratorState::valid,
303 ExcInternalError());
304 const unsigned int n_faces_agglomerated_neighbor =
305 neigh_polytope->n_faces();
306
307 // Loop over all cells of neighboring agglomerate
308 for (unsigned int f_out = 0; f_out < n_faces_agglomerated_neighbor;
309 ++f_out)
310 {
311 // Check if same master cell
312 if (neigh_polytope->neighbor(f_out).state() == IteratorState::valid)
313 if (neigh_polytope->neighbor(f_out)->index() == index())
314 return f_out;
315 }
316 return numbers::invalid_unsigned_int;
317 }
318 else
319 {
320 // Face is at boundary
321 return numbers::invalid_unsigned_int;
322 }
323}

333template <int dim, int spacedim>
334inline AgglomerationAccessor<dim, spacedim>::AgglomerationAccessor(
335 const typename Triangulation<dim, spacedim>::active_cell_iterator &cell,
336 const AgglomerationHandler<dim, spacedim> * ah)
337{
338 handler = const_cast<AgglomerationHandler<dim, spacedim> *>(ah);
339 if (&(*handler->master_cells_container.end()) == std::addressof(cell))
340 {
341 present_index = handler->master_cells_container.size();
342 master_cell = handler->master_cells_container[present_index];
343 }
344 else
345 {
346 present_index = handler->master2polygon.at(cell->active_cell_index());
347 master_cell = cell;
348 }
349}

389template <int dim, int spacedim>
390inline double
391AgglomerationAccessor<dim, spacedim>::diameter() const
392{
393 Assert(!handler->is_slave_cell(master_cell),
394 ExcMessage("The present function cannot be called for slave cells."));
395
396 if (handler->is_master_cell(master_cell))
397 {
398 // Get the bounding box associated with the master cell
399 const auto &bdary_pts =
400 handler->bboxes[present_index].get_boundary_points();
401 return (bdary_pts.second - bdary_pts.first).norm();
402 }
403 else
404 {
405 // Standard deal.II way to get the measure of a cell.
406 return master_cell->diameter();
407 }
408}

521AgglomerationAccessor<dim, spacedim>::n_agglomerated_faces() const
522{
523 const auto & agglomeration = get_agglomerate();
